70 double PolyFit::controlParameter(ParameterList& minPar){
71 
72  // Calculate chi^2 for current set of fit parameters
73  double result=0.;
74  for (unsigned int i=0; i<_xValue.size(); i++){
75  double yValFit = minPar.GetDoubleParameterValue(0)
76  +minPar.GetDoubleParameterValue(1)*_xValue[i]
77  +minPar.GetDoubleParameterValue(2)*_xValue[i]*_xValue[i]
78  +minPar.GetDoubleParameterValue(3)*_xValue[i]*_xValue[i]*_xValue[i];
79  double yValExp=_yValue[i];
80  double tmpChi=((yValExp-yValFit)*(yValExp-yValFit))/(_sigma*_sigma);
81  result+=tmpChi;
82  }
83  return result;
84 }
