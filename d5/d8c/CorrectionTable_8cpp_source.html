96 double CorrectionTable::GetValue(int charge, double momentum) const {
97  int binNumber = findBin(momentum);
98  if (binNumber < 0)
99  throw std::runtime_error(
100  "momentumSys::GetValue() no bin found for momentum value " +
101  std::to_string((long double)momentum) + "!");
102 
103  double val = 0.0;
104  if (charge > 0)
105  val = sys.at(binNumber);
106  else if (charge < 0)
107  val = antiSys.at(binNumber);
108  else if (charge == 0) { // charge unknown -> average value
109  val = (antiSys.at(binNumber) + sys.at(binNumber)) / 2;
110  }
111  return val;
112 }
113 double CorrectionTable::GetError(int charge, double momentum) const {
114  int binNumber = findBin(momentum);
115  if (binNumber < 0)
116  throw std::runtime_error(
117  "momentumSys::GetError() no bin found for momentum value " +
118  std::to_string((long double)momentum) + "!");
119 
120  double err = 0.0;
121  if (charge > 0) // D0->K0bar K+K-
122  err = sysError.at(binNumber);
123  else if (charge < 0) // D0->K0bar K+K-
124  err = antiSysError.at(binNumber);
125  else if (charge == 0) { // charge unknown -> average value
126  err = (0.5 * std::sqrt(antiSysError.at(binNumber) * antiSysError.at(binNumber) +
127  sysError.at(binNumber) * sysError.at(binNumber)));
128  }
129  return err;
130 }

142 double CorrectionTable::GetTotalSystematics(bool useArithmeticMean) {
143  double mean = 0;
144  double sumSigma = 0;
145  // use arithmetic mean
146  if (useArithmeticMean) {
147  for (unsigned int i = 0; i < totalSys.size(); i++)
148  mean += totalSys.at(i);
149  return mean / totalSys.size();
150  } else {
151  // use weighted mean
152  for (unsigned int i = 0; i < totalSys.size(); i++) {
153  double tmp = 1 / (totalSysError.at(i) * totalSysError.at(i));
154  mean += tmp * totalSys.at(i);
155  sumSigma += tmp;
156  }
157  mean /= sumSigma;
158  return mean;
159  }
160 }
161 double CorrectionTable::GetTotalSystematicsError(bool useArithmeticMean) {
162  double sumSigma = 0; // inverse sum over uncertainties
163  double mean = GetTotalSystematics(useArithmeticMean);
164  if (useArithmeticMean) {
165  for (unsigned int i = 0; i < totalSys.size(); i++)
166  sumSigma += (totalSys.at(i) - mean) * (totalSys.at(i) - mean);
167  return sumSigma = sqrt(sumSigma / totalSys.size());
168  } else {
169  for (unsigned int i = 0; i < totalSysError.size(); i++)
170  sumSigma += 1 / (totalSysError.at(i) * totalSysError.at(i));
171  return sqrt(1 / sumSigma);
172  }
173 }
