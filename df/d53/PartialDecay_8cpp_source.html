18 std::shared_ptr<Resonance>
19 PartialDecay::Factory(std::shared_ptr<PartList> partL,
20  std::shared_ptr<Kinematics> kin,
21  const boost::property_tree::ptree &pt) {
22 
23  LOG(trace) << "PartialDecay::Factory() |";
24  SubSystem subSys = SubSystemFactory(pt.get_child("SubSystem"));
25 
26  auto obj = std::make_shared<PartialDecay>(subSys);
27  obj->SetName(pt.get<std::string>("<xmlattr>.Name", "empty"));
28 
29  std::shared_ptr<DoubleParameter> mag, phase;
30  for (const auto &v : pt.get_child("")) {
31  if (v.first == "Parameter") {
32  if (v.second.get<std::string>("<xmlattr>.Type") == "Magnitude") {
33  auto tmp = ComPWA::DoubleParameterFactory(v.second);
34  mag = std::make_shared<DoubleParameter>(tmp);
35  }
36  if (v.second.get<std::string>("<xmlattr>.Type") == "Phase") {
37  auto tmp = ComPWA::DoubleParameterFactory(v.second);
38  phase = std::make_shared<DoubleParameter>(tmp);
39  }
40  } else {
41  // ignored further settings. Should we throw an error?
42  }
43  }
44 
45  if (mag)
46  obj->SetMagnitudeParameter(mag);
47  else
48  throw BadParameter(
49  "PartialDecay::Factory() | No magnitude parameter found.");
50 
51  if (phase)
52  obj->SetPhaseParameter(phase);
53  else
54  throw BadParameter(
55  "PartialDecay::Factory() | No phase parameter found.");
56 
57  auto dynObj = std::shared_ptr<DecayDynamics::AbstractDynamicalFunction>();
58  std::string name = pt.get<std::string>("DecayParticle.<xmlattr>.Name");
59 
60  // Two-body decay
61  if (subSys.GetFinalStates().size() == 2) {
62  // Create WignerD object
63  obj->SetWignerD(
64  ComPWA::Physics::HelicityFormalism::AmpWignerD::Factory(partL, pt));
65 
66  auto partProp = partL->find(name)->second;
67  std::string decayType = partProp.GetDecayType();
68 
69  if (decayType == "stable") {
70  throw std::runtime_error("PartialDecay::Factory() | Stable particle is "
71  "given as mother particle of a decay. Makes no "
72  "sense!");
73  } else if (decayType == "relativisticBreitWigner") {
74  dynObj = DecayDynamics::RelativisticBreitWigner::Factory(partL, pt);
75  } else if (decayType == "flatte") {
76  dynObj = DecayDynamics::AmpFlatteRes::Factory(partL, pt);
77  } else {
78  throw std::runtime_error("PartialDecay::Factory() | Unknown decay type " +
79  decayType + "!");
80  }
81 
82  // make sure dynamical function is created and set first
83  } else { // Multi-body decay
84  dynObj = std::shared_ptr<DecayDynamics::AbstractDynamicalFunction>(
85  new DecayDynamics::NonResonant);
86  dynObj->SetName(name);
87  // We assume the we have a multi-body decay and assume that the decay
88  // proceeds via constant (non-resonant) dynamics
89  obj->SetWignerD(std::shared_ptr<AmpWignerD>(new AmpWignerD()));
90  }
91 
92  obj->SetDynamicalFunction(dynObj);
93  obj->SetDataPosition(
94  3 *
95  std::dynamic_pointer_cast<HelicityKinematics>(kin)->GetDataID(subSys));
96 
97  obj->SetPhspVolume(kin->GetPhspVolume());
98 
99  return std::static_pointer_cast<Resonance>(obj);
100 }

138 double PartialDecay::GetNormalization() const {

163  tr->insertTree(_angD->GetTree(sample, _dataPos + 1, _dataPos + 2),
164  "Resonance(" + GetName() + ")" + suffix);
165  tr->insertTree(_dynamic->GetTree(sample, _dataPos),
166  "Resonance(" + GetName() + ")" + suffix);
167 
168  tr->recalculate();
169  tr->createNode("Normalization",
170  std::shared_ptr<Strategy>(new Inverse(ParType::DOUBLE)),
171  "Resonance(" + GetName() + ")" + suffix); // 1/normLH
172  tr->createNode("SqrtIntegral",
173  std::shared_ptr<Strategy>(new SquareRoot(ParType::DOUBLE)),
174  "Normalization");
175  tr->createNode("Integral",
176  std::shared_ptr<Strategy>(new MultAll(ParType::DOUBLE)),
177  "SqrtIntegral");
178  tr->createLeaf("PhspVolume", phspVolume_, "Integral");
179  tr->createLeaf("InverseSampleSize", 1 / ((double)phspSampleSize), "Integral");
180  tr->createNode("Sum", std::shared_ptr<Strategy>(new AddAll(ParType::DOUBLE)),
181  "Integral");
182  tr->createNode("Intensity",
183  std::shared_ptr<Strategy>(new AbsSquare(ParType::MDOUBLE)),
184  "Sum", phspSampleSize,
185  false); //|T_{ev}|^2
186  tr->createNode("mult",
187  std::shared_ptr<Strategy>(new MultAll(ParType::MCOMPLEX)),
188  "Intensity");
189  tr->insertTree(_angD->GetTree(toySample, _dataPos + 1, _dataPos + 2, "_norm"),
190  "mult" + suffix);
191  tr->insertTree(_dynamic->GetTree(toySample, _dataPos, "_norm"),
192  "mult" + suffix);
193 
194  tr->recalculate();
195  return tr;
196 }
