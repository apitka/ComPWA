171  void Boost(double __bx, double __by, double __bz) {
172 
173  // check to see if bx,by,bz are all less than 1
174  if (std::abs(__bx) >= 1 || std::abs(__by) >= 1 || std::abs(__bz) >= 1)
175  std::cout << "Error! Attempt to boost using invalid boost vector."
176  << std::endl;
177  assert((std::abs(__bx) < 1) && (std::abs(__by) < 1) &&
178  (std::abs(__bz) < 1));
179 
180  Tensor<double> lt(2); // Lorentz transformation tensor
181  double gamma = 1.0 / sqrt(1.0 - __bx * __bx - __by * __by - __bz * __bz);
182  double gamFact = (gamma * gamma) / (gamma + 1.0);
183 
184  // set up the Lorentz transformation tensor
185  lt.Zero();
186 
187  lt(0, 0) = gamma;
188  lt(0, 1) = gamma * __bx;
189  lt(0, 2) = gamma * __by;
190  lt(0, 3) = gamma * __bz;
191 
192  lt(1, 1) = (__bx * __bx * gamFact) + 1;
193  lt(1, 2) = __bx * __by * gamFact;
194  lt(1, 3) = __bx * __bz * gamFact;
195 
196  lt(2, 2) = (__by * __by * gamFact) + 1;
197  lt(2, 3) = __by * __bz * gamFact;
198 
199  lt(3, 3) = (__bz * __bz * gamFact) + 1;
200 
201  lt(1, 0) = lt(0, 1);
202  lt(2, 0) = lt(0, 2);
203  lt(2, 1) = lt(1, 2);
204  lt(3, 0) = lt(0, 3);
205  lt(3, 1) = lt(1, 3);
206  lt(3, 2) = lt(2, 3);
207 
208  this->Transform(lt);
209  }

222  void Rotate(double __alpha, double __beta, double __gamma) {
223 
224  double ca = cos(__alpha);
225  double sa = sin(__alpha);
226  double cb = cos(__beta);
227  double sb = sin(__beta);
228  double cg = cos(__gamma);
229  double sg = sin(__gamma);
230 
231  Tensor<double> lt(2); // Lorentz transformation tensor
232 
233  lt.Zero();
234 
235  lt(0, 0) = 1.0;
236 
237  lt(1, 1) = ca * cb * cg - sa * sg;
238  lt(1, 2) = sa * cb * cg + ca * sg;
239  lt(1, 3) = -sb * cg;
240 
241  lt(2, 1) = -ca * cb * sg - sa * cg;
242  lt(2, 2) = -sa * cb * sg + ca * cg;
243  lt(2, 3) = sb * sg;
244 
245  lt(3, 1) = ca * sb;
246  lt(3, 2) = sa * sb;
247  lt(3, 3) = cb;
248 
249  this->Transform(lt);

512  template <typename T>
513  Tensor<typename MultType<_Tp, T>::Type>
514  operator%(const Tensor<T> &__tensor) const {
515 
516  int rank = this->_rank + __tensor._rank;
517  Tensor<typename MultType<_Tp, T>::Type> ret(rank);
518 
519  // if either tensor is rank 0, just return this*__tensor
520  if ((_rank == 0) || (__tensor._rank == 0))
521  return (*this) * __tensor;
522  else { // we actually have to do some work
523  TensorIndex index(rank);
524  TensorIndex ind1(this->_rank);
525  TensorIndex ind2(__tensor._rank);
526 
527  int size1 = ind1.Size();
528  // int size2 = ind2.Size();
529 
530  while (index.IsValid()) { // loop over ret's elements
531 
532  // set up this' indicies
533  for (int i = 0; i < size1; i++)
534  ind1.SetIndex(i, index[i]);
535  // set up _tensor's indicies
536  for (int i = size1; i < index.Size(); i++)
537  ind2.SetIndex(i - size1, index[i]);
538

540  ret(index) = (this->Element(ind1)) * (__tensor(ind2));
541  index++;
542  }
543  }

726  template <typename T>
727  Tensor<typename MultType<_Tp, T>::Type>
728  Contract(const Tensor<T> &__tensor, int __num_indicies = -1) const {
729 
730  int ind1max, ind2st, sumsize, nterm, rank;
731  Tensor<typename MultType<_Tp, T>::Type> ret;
732  // MetricTensor g;
733  Tensor<double> g(2); // Metric tensor
734  g.Element(0, 0) = 1.0;
735  g.Element(1, 1) = -1.0;
736  g.Element(2, 2) = -1.0;
737  g.Element(3, 3) = -1.0;
738 
739  double gFactors;
740  typename MultType<_Tp, T>::Type element;
741  ind1max = 0;
742  ind2st = 0;
743  sumsize = 0;
744 
745  if (_rank ==
746  0) { // if this is rank 0, just multiply __tensor by this' value
747  ret.SetRank(__tensor._rank);
748  ret = _data[0] * __tensor;
749  return ret;
750  } else if (__tensor._rank == 0) { //__tensor is rank 0...
751  ret.SetRank(_rank);
752  ret = (*this) * __tensor._data[0];
753  return ret;
754  }
755 
756  if (__num_indicies > _rank || __num_indicies > __tensor._rank) {
757  std::cout << "<Tensor::Contract> Error! Can't contract " << __num_indicies
758  << " between a " << _rank << " rank and a " << __tensor._rank
759  << " rank tensor." << std::endl;
760  abort();
761  }
762 
763  if (__num_indicies > 0)
764  rank = _rank + __tensor._rank - 2 * __num_indicies;
765  else
766  rank = abs(_rank - __tensor._rank);
767 
768  // set ret's rank and create the summed TensorIndex
769  ret.SetRank(rank);
770  TensorIndex indSummed;
771 
772  if (__num_indicies < 0) {
773  // check to see which tensor has the smaller rank (calculate how many
774  // summed indicies are needed)
775  if (_rank <= __tensor._rank)
776  sumsize = _rank;
777  else
778  sumsize = __tensor._rank;
779  } else
780  sumsize = __num_indicies;
781  indSummed.Resize(2 * sumsize);
782 
783  TensorIndex ind1(this->_rank);
784  TensorIndex ind2(__tensor._rank);
785 
786  int size1 = ind1.Size();
787  int size2 = ind2.Size();
788 
789  if (rank > 0) { // rank > 0, we need to do all the loops
790  TensorIndex index(rank);
791 
792  while (index.IsValid()) { // loop over ret's elements
793 
794  // check to see if this will have any free indicies
795  if ((size1 - sumsize) > 0)
796  ind1max = size1 - sumsize;
797  else
798  ind1max = 0;
799 
800  // set this index (except last ??(number of summed indicies) indicies)
801  for (int i = 0; i < ind1max; i++)
802  ind1.SetIndex(i, index[i]);
803 
804  // set __tensor index (except 1st ??(summed indicies) indicies)
805  ind2st = sumsize;
806  for (int i = ind2st; i < size2; i++)
807  ind2.SetIndex(i, index[ind1max + (i - ind2st)]);
808 
809  nterm = 0;
810  while (indSummed.IsValid()) { // loop over summed indicies
811 
812  gFactors = g(indSummed[0], indSummed[0 + indSummed.Size() / 2]);
813  // get the needed amount of metric tensor factors
814  for (int i = 1; i < indSummed.Size() / 2; i++) {
815  gFactors *= g(indSummed[i], indSummed[i + indSummed.Size() / 2]);
816  }
817  if (gFactors != 0.0) {
818  nterm++;
819  // set up last ?? this and 1st ?? __tensor indicies
820  for (int i = ind1max; i < size1; i++)
821  ind1.SetIndex(i, indSummed[i - ind1max]);
822  for (int i = size1 - ind1max; i < indSummed.Size(); i++)
823  ind2.SetIndex(i - (size1 - ind1max), indSummed[i]);
824 
825  // multiply the metric tensor factor by this and __tensor elements
826  element = (this->Element(ind1)) * (__tensor(ind2)) * gFactors;
827 
828  // add to each element this*__tensor*g*g...*g with correct # of g's
829  if (nterm == 1)
830  ret(index) = element;
831  else
832  ret(index) += element;
833  }
834  ++indSummed;
835  }
836  // reset the summed indicies, step up index to next ret element
837  indSummed.Reset();
838  ++index;
839  }
840  } else { // both are same rank tensors (R is rank 0)
841  nterm = 0;
842 
843  // loop over summed indicies (only loop needed in this case)
844  while (indSummed.IsValid()) {
845  gFactors = g(indSummed[0], indSummed[0 + indSummed.Size() / 2]);
846 
847  // get the needed amount of metric tensor factors
848  for (int i = 1; i < indSummed.Size() / 2; i++)
849  gFactors *= g(indSummed[i], indSummed[i + indSummed.Size() / 2]);
850 
851  if (gFactors != 0.0) {
852  nterm++;
853  for (int i = 0; i < indSummed.Size() / 2; i++) {

942  void Transform(const Tensor<double> &__lt) {
943 
944  if (__lt.Rank() != 2)
945  std::cout << "Error! Lorentz transformation tensor NOT rank 2."
946  << std::endl;
947  assert(__lt.Rank() == 2);
948  int rank = this->Rank();
949  if (rank > 0) { // if rank 0 no transformation needed
950  TensorIndex index(rank);
951  TensorIndex indSummed(rank);
952  int nterm;
953  double lamFact;
954  // make a copy
955  Tensor<_Tp> copy(*this);
956 
957  while (index.IsValid()) { // loop over elements of this tensor
958  nterm = 0;
959  while (indSummed.IsValid()) {
960  // get the appropriate number of Lambda_mu_nu factors
961  lamFact = __lt(index[0], indSummed[0]);
962  for (int i = 1; i < rank; i++)
963  lamFact *= __lt(index[i], indSummed[i]);
964  if (lamFact != 0.0) {
965  nterm++;
966  // add to each element this*Lambda*Lambda*...*Lambda
967  if (nterm == 1)
968  (*this)(index) = lamFact * (copy(indSummed));
969  else
970  (*this)(index) += lamFact * (copy(indSummed));
971  }
972  ++indSummed;
973  }
974  // reset summed indicies, step up index to next element
975  indSummed.Reset();
976  ++index;
977  }
978  }
979  }
