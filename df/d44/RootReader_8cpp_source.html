55 void RootReader::read(TTree *fTree, double readSize) {
56 
57  // TTree branch variables
58  TClonesArray *fParticles;
59  double feventWeight;
60  double feventEff;
61  int fCharge;
62  int fFlavour;
63 
64  fParticles = new TClonesArray("TParticle");
65  fTree->GetBranch("Particles")->SetAutoDelete(false);
66  fTree->SetBranchAddress("Particles", &fParticles);
67  fTree->SetBranchAddress("eff", &feventEff);
68  fTree->SetBranchAddress("weight", &feventWeight);
69  fTree->SetBranchAddress("charge", &fCharge);
70  fTree->SetBranchAddress("flavour", &fFlavour);
71 
72  if (readSize <= 0 || readSize > fTree->GetEntries())
73  readSize = fTree->GetEntries();
74 
75  for (unsigned int evt = 0; evt < readSize; evt++) {
76  Event tmp;
77  fParticles->Clear();
78  fTree->GetEntry(evt);
79 
80  // Get number of particle in TClonesrray
81  unsigned int nParts = fParticles->GetEntriesFast();
82 
83  TParticle *partN;
84  TLorentzVector inN;
85  for (unsigned int part = 0; part < nParts; part++) {
86  partN = 0;
87  partN = (TParticle *)fParticles->At(part);
88  if (!partN)
89  continue;
90  partN->Momentum(inN);
91  int charge = partN->GetPDG()->Charge();
92  if (charge != 0)
93  charge /= std::fabs(charge);
94  tmp.AddParticle(Particle(inN.X(), inN.Y(), inN.Z(), inN.E(),
95  partN->GetPdgCode(), charge));
96  } // particle loop
97  tmp.SetWeight(feventWeight);
98  tmp.SetCharge(fCharge);
99  tmp.SetFlavour(fFlavour);
100  tmp.SetEfficiency(feventEff);
101 
102  fEvents.push_back(tmp);
103  if (feventWeight > maxWeight)
104  maxWeight = feventWeight;
105  } // end event loop
106 }

134  TClonesArray &partArray = *fParticles;
135 
136  auto it = fEvents.begin();
137  for (; it != fEvents.end(); ++it) {
138  fParticles->Clear();
139  feventWeight = (*it).GetWeight();
140  fCharge = (*it).GetCharge();
141  fFlavour = (*it).GetFlavour();
142  feventEff = (*it).GetEfficiency();
143 
144  TLorentzVector motherMomentum(0, 0, 0, (*it).GetCMSEnergy());
145  for (unsigned int i = 0; i < numPart; i++) {
146  const Particle oldParticle = (*it).GetParticle(i);
147  TLorentzVector oldMomentum(oldParticle.GetPx(), oldParticle.GetPy(),
148  oldParticle.GetPz(), oldParticle.GetE());
149  new (partArray[i]) TParticle(oldParticle.GetPid(), 1, 0, 0, 0, 0,
150  oldMomentum, motherMomentum);
151  }
152  fTree->Fill();
153  }

155  fFile->Close();
