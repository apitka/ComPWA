95 void RootReader::storeEvents() {
96  if (readSize <= 0 || readSize > fTree->GetEntries())
97  readSize = fTree->GetEntries();
98 
99  for (unsigned int evt = 0; evt < readSize; evt++) {
100  Event tmp;
101  fParticles->Clear();
102  fTree->GetEntry(evt);
103 
104  // Get number of particle in TClonesrray
105  unsigned int nParts = fParticles->GetEntriesFast();
106 
107  TParticle *partN;
108  TLorentzVector inN;
109  for (unsigned int part = 0; part < nParts; part++) {
110  partN = 0;
111  partN = (TParticle *)fParticles->At(part);
112  if (!partN)
113  continue;
114  partN->Momentum(inN);
115  int charge = partN->GetPDG()->Charge();
116  if (charge != 0)
117  charge /= std::fabs(charge);
118  tmp.AddParticle(Particle(inN.X(), inN.Y(), inN.Z(), inN.E(),
119  partN->GetPdgCode(), charge));
120  } // particle loop
121  tmp.SetWeight(feventWeight);
122  tmp.SetCharge(fCharge);
123  tmp.SetFlavour(fFlavour);
124  tmp.SetEfficiency(feventEff);
125 
126  fEvents.push_back(tmp);
127  if (feventWeight > maxWeight)
128  maxWeight = feventWeight;
129  } // event loop
130 }

132 void RootReader::WriteData(std::string file, std::string trName) {
133  if (file != "")
134  fileName = file;
135  if (trName != "")
136  treeName = trName;

138  LOG(info) << "RootReader::writeData() | Writing current "

141 
142  TFile *fFile = new TFile(fileName.c_str(), "RECREATE");

144  throw std::runtime_error("RootReader::WriteData() | Can't open data file: "+
145  fileName);
146 
147  fTree = new TTree(treeName.c_str(), treeName.c_str());
148  unsigned int numPart = fEvents.at(0).GetNParticles();
149  fParticles = new TClonesArray("TParticle", numPart);
150  fTree->Branch("Particles", &fParticles);
151  fTree->Branch("weight", &feventWeight, "weight/D");
152  fTree->Branch("eff", &feventEff, "weight/D");
153  fTree->Branch("charge", &fCharge, "charge/I");
154  fTree->Branch("flavour", &fFlavour, "flavour/I");
155  TClonesArray &partArray = *fParticles;
156 
157  auto it = fEvents.begin();
158  for (; it != fEvents.end(); ++it) {
159  fParticles->Clear();
160  feventWeight = (*it).GetWeight();
161  fCharge = (*it).GetCharge();
162  fFlavour = (*it).GetFlavour();
163  feventEff = (*it).GetEfficiency();
164 
165  TLorentzVector motherMomentum(0, 0, 0, (*it).GetCMSEnergy());
166  for (unsigned int i = 0; i < numPart; i++) {
167  const Particle oldParticle = (*it).GetParticle(i);
168  TLorentzVector oldMomentum(oldParticle.GetPx(), oldParticle.GetPy(),
169  oldParticle.GetPz(), oldParticle.GetE());
170  new (partArray[i]) TParticle(oldParticle.GetPid(), 1, 0, 0, 0, 0,
171  oldMomentum, motherMomentum);
172  }
173  fTree->Fill();
174  }
175  fTree->Write("", TObject::kOverwrite, 0);
176  fFile->Close();
177  return;

180 void RootReader::bin() {
181  double min = -1, max = -1;
182  fmaxBins = 500; // TODO setter, consturctor
183  TLorentzVector in1, in2;
184 
185  // initialize min & max
186  fTree->GetEntry(0);
187  unsigned int nParts = fParticles->GetEntriesFast();
188  if (nParts != 2)
189  return;
190  TParticle *part1 = (TParticle *)fParticles->At(0); // pip
191  TParticle *part2 = (TParticle *)fParticles->At(1); // pim
192  if (!part1 || !part2)
193  return;
194  part1->Momentum(in1);
195  part2->Momentum(in2);
196  double inm12 = (in1 + in2).Mag2();
197  min = max = inm12;
198 
199  // find min and max
200  for (unsigned int evt = 1; evt < fTree->GetEntries(); evt++) {
201  fTree->GetEntry(evt);
202  unsigned int nParts = fParticles->GetEntriesFast();
203  if (nParts != 2)
204  return;
205 
206  part1 = (TParticle *)fParticles->At(0); // pip
207  part2 = (TParticle *)fParticles->At(1); // pim
208  if (!part1 || !part2)
209  return;
210  part1->Momentum(in1);
211  part2->Momentum(in2);
212 
213  inm12 = (in1 + in2).Mag2();
214 
215  if (min > inm12)
216  min = inm12;
217  if (max < inm12)
218  max = inm12;
219  }
220 
221  // initialize bins with weight zero
222  min = sqrt(min);
223  max = sqrt(max);
224  double step = (max - min) / (double)fmaxBins;
225  for (unsigned int bin = 0; bin < fmaxBins; bin++) {
226  fBins[bin] = std::make_pair(min + bin * step, 0.0);
227  }
228 
229  // fill bins
230  for (unsigned int evt = 0; evt < fmaxBins; evt++) {
231  fTree->GetEntry(evt);
232  unsigned int nParts = fParticles->GetEntriesFast();
233  if (nParts != 2)
234  return;
235 
236  TParticle *part1 = (TParticle *)fParticles->At(0); // pip
237  TParticle *part2 = (TParticle *)fParticles->At(1); // pim
238  if (!part1 || !part2)
239  return;
240  part1->Momentum(in1);
241  part2->Momentum(in2);
242  inm12 = sqrt((in1 + in2).Mag2());
243 
244  int bin = (int)((inm12 - min) / step);
245  fBins[bin].second += 1.;
246  }
247 }
