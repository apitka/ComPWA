400 void DoubleParameter::UpdateParameter(std::shared_ptr<DoubleParameter> newPar) {
401 
402  // Copy bounds
403  if (newPar->HasBounds()){
404  bounds_ = 1;
405  SetMinMax(newPar->GetMinValue(), newPar->GetMaxValue());
406  } else
407  bounds_ = 0;
408 
409  bool isFix = newPar->IsFixed();
410  FixParameter(0); // we ignore here if parameter is fixed
411 
412  // Copy value
413  SetValue(newPar->GetValue());
414 
415  // Copy uncertainty
416  if (newPar->GetErrorType() == ErrorType::SYM)
417  SetError(newPar->GetError());
418  else if (newPar->GetErrorType() == ErrorType::ASYM)
419  SetError(newPar->GetErrorLow(), newPar->GetErrorHigh());
420  else
421  SetErrorType(ErrorType::NOTDEF);
422 
423  // Copy fix parameter
424  FixParameter(isFix);
425  return;
426 }

729 bool IntegerParameter::operator==(const IntegerParameter otherPar) const {
730  if (this->type() != otherPar.type())
731  return false;
732  if (this->GetName() != otherPar.GetName())
733  return false;
734  if (this->GetValue() != otherPar.GetValue())
735  return false;
736 
737  /* We assume that if name and value are the same both parameters match. In
738  * case that other properties of the parameterts differ we throw an
739  * exception since we assume that this is a user mistake.
740  */
741  if (bounds_ != otherPar.bounds_ || usebounds_ != otherPar.usebounds_ ||
742  this->GetMinValue() != otherPar.GetMinValue() ||
743  this->GetMaxValue() != otherPar.GetMaxValue())
744  throw std::runtime_error("IntegerParameter::operator==() | Parameters "
745  "match by name (" +
746  GetName() + ") and value (" +
747  std::to_string(GetValue()) +
748  ") but differs in "
749  "parameter bounds. We assume that there is a "
750  "mistake. Check your input files!");
751 
752  if (fixed_ != otherPar.fixed_)
753  throw std::runtime_error(
754  "IntegerParameter::operator==() | Parameters "
755  "match by name (" +
756  GetName() + ") and value (" + std::to_string(GetValue()) +
757  ") but one is fixed the other not. "
758  "We assume that there is a mistake. Check your input files!");
759 
760  if (err_ != otherPar.err_)
761  throw std::runtime_error("IntegerParameter::operator==() | Parameters "
762  "match by name (" +
763  GetName() + ") and value (" +
764  std::to_string(GetValue()) +
765  ") but differs in "
766  "parameter error. We assume that there is a "
767  "mistake. Check your input files!");
768 
769  return true;
770 }
