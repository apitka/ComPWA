82 inline void multivariateGaussian(const gsl_rng *rnd, const int vecSize,
83  const gsl_vector *in, const gsl_matrix *cov,
84  gsl_vector *res) {
85  // Generate and fill temporary covariance matrix
86  gsl_matrix *tmpM = gsl_matrix_alloc(vecSize, vecSize);
87  gsl_matrix_memcpy(tmpM, cov);
88 
89  // Cholesky decomposition
90  int status = gsl_linalg_cholesky_decomp(tmpM);
91  if (status == GSL_EDOM)
92  LOG(error) << "Decomposition has failed!";
93 
94  // Compute vector of random gaussian variables
95  for (unsigned int i = 0; i < vecSize; i++)
96  gsl_vector_set(res, i, gsl_ran_ugaussian(rnd));
97 
98  // Debug
99  // gsl_matrix_print(cov);
100  // gsl_vector_print(in);
101  // gsl_vector_print(res);
102  // gsl_matrix_print(tmpM);
103 
104  /*From the GNU GSL Documentation:
105  * The function dtrmv compute the matrix-vector product x = op(A) x for the
106  * triangular matrix A, where op(A) = A, A^T, A^H for TransA = CblasNoTrans,
107  * CblasTrans, CblasConjTrans. When Uplo is CblasUpper then the upper
108  * triangle of A is used, and when Uplo is CblasLower then the lower
109  * triangle of A is used. If Diag is CblasNonUnit then the diagonal of
110  * the matrix is used, but if Diag is CblasUnit then the diagonal elements
111  * of the matrix A are taken as unity and are not referenced.
112  */
113  gsl_blas_dtrmv(CblasLower, CblasNoTrans, CblasNonUnit, tmpM, res);
114 
115  gsl_vector_add(res, in);
116  // free temporary object
117  gsl_matrix_free(tmpM);
118 };

120 inline ComPWA::DoubleParameter
121 CalculateFitFraction(std::shared_ptr<ComPWA::Kinematics> kin,
122  std::shared_ptr<ComPWA::AmpIntensity> intens,
123  std::shared_ptr<std::vector<dataPoint>> sample,
124  const std::pair<std::string, std::string> def) {
125 
126  std::shared_ptr<ComPWA::AmpIntensity> numer, denom;
127  if (def.first == intens->Name())
128  numer = intens;
129  else
130  numer = intens->GetComponent(def.first);
131 
132  if (def.second == intens->Name())
133  denom = intens;
134  else
135  denom = intens->GetComponent(def.second);
136 
137  double phspVolume = kin->GetPhspVolume();
138 
139  double integral_numerator =
140  ComPWA::Tools::Integral(numer, sample, phspVolume);
141  double integral_denominator =
142  ComPWA::Tools::Integral(denom, sample, phspVolume);
143  double ffVal = integral_numerator / integral_denominator;
144  LOG(trace) << "CalculateFitFraction() | Result for (" << def.first << "/"
145  << def.second << ") is " << integral_numerator << "/"
146  << integral_denominator << "=" << ffVal;
147  return DoubleParameter(def.first, ffVal, 0.0);
148 }
