35 void MinuitResult::init(ROOT::Minuit2::FunctionMinimum min) {
36  ROOT::Minuit2::MnUserParameterState minState = min.UserState();
37 
38  if (minState.HasCovariance()) {
39  ROOT::Minuit2::MnUserCovariance minuitCovMatrix = minState.Covariance();
40  /* Size of Minuit covariance vector is given by dim*(dim+1)/2.
41  * dim is the dimension of the covariance matrix.
42  * The dimension can therefore be calculated as
43  * dim = -0.5+-0.5 sqrt(8*size+1)
44  */
45  nFreeParameter = minuitCovMatrix.Nrow();
46  globalCC = minState.GlobalCC().GlobalCC();
47  cov = std::vector<std::vector<double>>(nFreeParameter,
48  std::vector<double>(nFreeParameter));
49  corr = std::vector<std::vector<double>>(
50  nFreeParameter, std::vector<double>(nFreeParameter));
51  for (unsigned i = 0; i < nFreeParameter; ++i)
52  for (unsigned j = i; j < nFreeParameter; ++j) {
53  cov.at(i).at(j) = minuitCovMatrix(j, i);
54  cov.at(j).at(i) = minuitCovMatrix(j, i); // fill lower half
55  }
56  for (unsigned i = 0; i < nFreeParameter; ++i)
57  for (unsigned j = i; j < nFreeParameter; ++j) {
58  corr.at(i).at(j) =
59  cov.at(i).at(j) / sqrt(cov.at(i).at(i) * cov.at(j).at(j));
60  corr.at(j).at(i) = corr.at(i).at(j); // fill lower half
61  }
62 
63  } else
64  LOG(error) << "MinuitResult: no valid correlation matrix available!";
65  initialLH = -1;
66  finalLH = minState.Fval();
67  edm = minState.Edm();
68  isValid = min.IsValid();
69  covPosDef = min.HasPosDefCovar();
70  hasValidParameters = min.HasValidParameters();
71  hasValidCov = min.HasValidCovariance();
72  hasAccCov = min.HasAccurateCovar();
73  hasReachedCallLimit = min.HasReachedCallLimit();
74  edmAboveMax = min.IsAboveMaxEdm();
75  hesseFailed = min.HesseFailed();
76  errorDef = min.Up();
77  nFcn = min.NFcn();
78 
79  return;
80 }

188 void MinuitResult::PrintCorrelationMatrix(TableFormater *tableCorr) {
189  if (!hasValidCov)
190  return;
191  tableCorr->addColumn(" ", 15); // add empty first column
192  tableCorr->addColumn("GlobalCC", 10); // global correlation coefficient
193 
194  // add columns in correlation matrix
195  for (unsigned int o = 0; o < finalParameters.GetNDouble(); o++) {
196  std::shared_ptr<DoubleParameter> ppp =
197  finalParameters.GetDoubleParameter(o);
198  if (ppp->IsFixed())
199  continue;
200  tableCorr->addColumn(ppp->GetName(), 15);
201  }
202 
203  unsigned int n = 0;
204  tableCorr->header();
205  for (unsigned int o = 0; o < finalParameters.GetNDouble(); o++) {
206  std::shared_ptr<DoubleParameter> ppp =
207  finalParameters.GetDoubleParameter(o);
208  if (ppp->IsFixed())
209  continue;
210  *tableCorr << ppp->GetName();
211  *tableCorr << globalCC.at(n);
212  for (unsigned int t = 0; t < corr.size(); t++) {
213  if (n >= corr.at(0).size()) {
214  *tableCorr << " ";
215  continue;
216  }
217  if (t >= n)
218  *tableCorr << corr.at(n).at(t);
219  else
220  *tableCorr << "";
221  }
222  n++;
223  }
224  tableCorr->footer();
225  return;
226 }
