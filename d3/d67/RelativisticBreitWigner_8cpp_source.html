40  obj->SetFormFactorType(ffType);
41 
42  // Read parameters from tree. Currently parameters of type 'Width' and
43  // 'MesonRadius' are required.
44  for (const auto &v : decayTr.get_child("")) {
45  if (v.first != "Parameter")
46  continue;
47  std::string type = v.second.get<std::string>("<xmlattr>.Type");
48  if (type == "Width") {
49  auto width = ComPWA::DoubleParameterFactory(v.second);
50  obj->SetWidthParameter(std::make_shared<DoubleParameter>(width));
51  } else if (type == "MesonRadius") {
52  auto mesonRadius = ComPWA::DoubleParameterFactory(v.second);
53  obj->SetMesonRadiusParameter(
54  std::make_shared<DoubleParameter>(mesonRadius));
55  } else {
56  throw std::runtime_error(
57  "RelativisticBreitWigner::Factory() | Parameter of type " + type +
58  " is unknown.");
59  }
60  }
61 
62  // Get masses of decay products
63  auto decayProducts = pt.get_child("SubSystem.DecayProducts");
64  if (decayProducts.size() != 2)
65  throw boost::property_tree::ptree_error(
66  "RelativisticBreitWigner::Factory() | Expect exactly two decay "

68  std::to_string(decayProducts.size()) + " given)!");
69 
70  auto firstItr = decayProducts.begin();
71  // auto secondItr = decayProducts.begin()+1; //compile error, no idea for why
72  auto secondItr = decayProducts.begin();
73  secondItr++;
74 
75  std::pair<std::string, std::string> daughterNames(

77  secondItr->second.get<std::string>("<xmlattr>.Name"));
78  std::pair<double, double> daughterMasses(

80  partL->find(daughterNames.second)->second.GetMass());
81  obj->SetDecayMasses(daughterMasses);

87  << daughterNames.second;

115 std::complex<double> RelativisticBreitWigner::dynamicalFunction(
116  double mSq, double mR, double ma, double mb, double width, unsigned int J,
117  double mesonRadius, formFactorType ffType) {
118 
119  std::complex<double> i(0, 1);
120  double sqrtS = sqrt(mSq);
121 
122  auto phspFactorSqrtS = phspFactor(sqrtS, ma, mb);
123  auto phspFactormR = phspFactor(mR, ma, mb);
124 
125  // Check if we have an event which is exactly at the phase space boundary
126  if (phspFactorSqrtS == std::complex<double>(0, 0))
127  return std::complex<double>(0, 0);
128 
129  std::complex<double> qTerm =
130  std::pow((phspFactorSqrtS / phspFactormR) * mR / sqrtS, (2 * J + 1));
131  double barrier = FormFactor(sqrtS, ma, mb, J, mesonRadius, ffType) /
132  FormFactor(mR, ma, mb, J, mesonRadius, ffType);
133 
134  // Calculate coupling constant to final state
135  std::complex<double> g_final =
136  widthToCoupling(mSq, mR, width, ma, mb, J, mesonRadius, ffType);
137 
138  // Coupling constant from production reaction. In case of a particle decay
139  // the production coupling doesn't depend in energy since the CM energy
140  // is in the (RC) system fixed to the mass of the decaying particle
141  double g_production = 1;
142 
143  std::complex<double> denom = std::complex<double>(mR * mR - mSq, 0) +
144  (-1.0) * i * sqrtS * (width * qTerm * barrier);
145 
146  std::complex<double> result = g_final * g_production / denom;
147 
148  assert(
149  (!std::isnan(result.real()) || !std::isinf(result.real())) &&
150  "RelativisticBreitWigner::dynamicalFunction() | Result is NaN or Inf!");
151  assert(
152  (!std::isnan(result.imag()) || !std::isinf(result.imag())) &&
153  "RelativisticBreitWigner::dynamicalFunction() | Result is NaN or Inf!");
154 
155  return result;
156 }

182 bool BreitWignerStrategy::execute(ParameterList &paras,
183  std::shared_ptr<AbsParameter> &out) {
184 #ifndef NDEBUG
185  // Check parameter type
186  if (checkType != out->type())
187  throw(WrongParType("BreitWignerStrat::execute() | "
188  "Output parameter is of type " +
189  std::string(ParNames[out->type()]) +
190  " and conflicts with expected type " +
191  std::string(ParNames[checkType])));
192 
193  // How many parameters do we expect?
194  int check_nBool = 0;
195  int check_nInt = 0;
196  int check_nComplex = 0;
197  int check_nDouble = 7;
198  int check_nMDouble = 1;
199  int check_nMComplex = 0;
200 
201  // Check size of parameter list
202  if (paras.GetNBool() != check_nBool)
203  throw(BadParameter("BreitWignerStrat::execute() | "
204  "Number of BoolParameters does not match: " +
205  std::to_string(paras.GetNBool()) + " given but " +
206  std::to_string(check_nBool) + " expected."));
207  if (paras.GetNInteger() != check_nInt)
208  throw(BadParameter("BreitWignerStrat::execute() | "
209  "Number of IntParameters does not match: " +
210  std::to_string(paras.GetNInteger()) + " given but " +
211  std::to_string(check_nInt) + " expected."));
212  if (paras.GetNDouble() != check_nDouble)
213  throw(BadParameter("BreitWignerStrat::execute() | "
214  "Number of DoubleParameters does not match: " +
215  std::to_string(paras.GetNDouble()) + " given but " +
216  std::to_string(check_nDouble) + " expected."));
217  if (paras.GetNComplex() != check_nComplex)
218  throw(BadParameter("BreitWignerStrat::execute() | "
219  "Number of ComplexParameters does not match: " +
220  std::to_string(paras.GetNComplex()) + " given but " +
221  std::to_string(check_nComplex) + " expected."));
222  if (paras.GetNMultiDouble() != check_nMDouble)
223  throw(BadParameter("BreitWignerStrat::execute() | "
224  "Number of MultiDoubles does not match: " +
225  std::to_string(paras.GetNMultiDouble()) + " given but " +
226  std::to_string(check_nMDouble) + " expected."));
227  if (paras.GetNMultiComplex() != check_nMComplex)
228  throw(BadParameter("BreitWignerStrat::execute() | "
229  "Number of MultiComplexes does not match: " +
230  std::to_string(paras.GetNMultiComplex()) +
231  " given but " + std::to_string(check_nMComplex) +
232  " expected."));
233 #endif
234 
239  double m0 = paras.GetDoubleParameter(0)->GetValue();
240  double Gamma0 = paras.GetDoubleParameter(1)->GetValue();
241  unsigned int spin = (unsigned int)paras.GetDoubleParameter(2)->GetValue();
242  double d = paras.GetDoubleParameter(3)->GetValue();
243  formFactorType ffType =
244  formFactorType(paras.GetDoubleParameter(4)->GetValue());
245  double ma = paras.GetDoubleParameter(5)->GetValue();
246  double mb = paras.GetDoubleParameter(6)->GetValue();
247 
248  std::vector<double> mp = paras.GetMultiDouble(0)->GetValues();
249 
250  std::vector<std::complex<double>> results(mp.size(),
251  std::complex<double>(0., 0.));
252 
253  // calc function for each point
254  for (unsigned int ele = 0; ele < mp.size(); ele++) {
255  try {
256  results.at(ele) = RelativisticBreitWigner::dynamicalFunction(
257  mp.at(ele), m0, ma, mb, Gamma0, spin, d, ffType);
258  } catch (std::exception &ex) {
259  LOG(error) << "BreitWignerStrategy::execute() | " << ex.what();
260  throw(std::runtime_error("BreitWignerStrategy::execute() | "
261  "Evaluation of dynamic function failed!"));
262  }
263  }
264  out =
265  std::shared_ptr<AbsParameter>(new MultiComplex(out->GetName(), results));
266  return true;
267 }

269 void RelativisticBreitWigner::GetParameters(ParameterList &list) {
270  AbstractDynamicalFunction::GetParameters(list);
271 
272  // We check of for each parameter if a parameter of the same name exists in
273  // list. If so we check if both are equal and set the local parameter to the
274  // parameter from the list. In this way we connect parameters that occur on
275  // different positions in the amplitude.
276  std::shared_ptr<DoubleParameter> tmp, width, radius;
277  width = GetWidthParameter();
278  radius = GetMesonRadiusParameter();
279  try { // catch BadParameter

281  // catch and throw std::runtime_error due to failed parameter comparisson
282  try {
283  if (*tmp == *width)
284  SetWidthParameter(tmp);
285  } catch (std::exception &ex) {
286  throw;
287  }
288  } catch (BadParameter &ex) {
289  list.AddParameter(width);
290  }
291 
292  try { // catch BadParameter
293  tmp = list.GetDoubleParameter(radius->GetName());
294  // catch and throw std::runtime_error due to failed parameter comparisson
295  try {
296  if (*tmp == *radius)
297  SetMesonRadiusParameter(tmp);

300  }
