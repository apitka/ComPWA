56 void TreeNode::recalculate() {
57  if (_children.size() < 1) {
58  _changed = false;
59  return;
60  }
61 
62  if (_value.size() == 1) { // i have just one dim, merge everything
63  ParameterList newVals;
64  for (unsigned int i = 0; i < _children.size(); i++) { // all children
65  for (unsigned int ele = 0; ele < _children.at(i)->getDim();

68  // LOG(debug) <<"TreeNode::recalculate() |

72  std::shared_ptr<AbsParameter> para = _children.at(i)->getValue(ele);

76  // para->type();

88  } else { // i have a certain dim, children must fill it

204 std::complex<double> TreeNode::getChildSingleValue(std::string name) const {
205  std::shared_ptr<TreeNode> node = std::shared_ptr<TreeNode>();
206  node = getChildNode(name);
207  if (node) {
208  std::shared_ptr<AbsParameter> val = node->getValue();
209  if (val->type() == ParType::DOUBLE)
210  return std::complex<double>(
211  (std::dynamic_pointer_cast<DoubleParameter>(val))->GetValue(), 0);
212  if (val->type() == ParType::COMPLEX)
213  return std::complex<double>(
214  (std::dynamic_pointer_cast<ComplexParameter>(val))->GetValue());
215  if (val->type() == ParType::INTEGER)
216  return std::complex<double>(
217  (std::dynamic_pointer_cast<IntegerParameter>(val))->GetValue(), 0);
218  if (val->type() == ParType::BOOL)
219  return std::complex<double>(
220  (std::dynamic_pointer_cast<BoolParameter>(val))->GetValue(), 0);
221  if (val->type() == ParType::MDOUBLE)
222  return std::complex<double>(
223  (std::dynamic_pointer_cast<MultiDouble>(val))->GetValue(0), 0);
224  if (val->type() == ParType::MCOMPLEX)
225  return std::complex<double>(
226  (std::dynamic_pointer_cast<MultiComplex>(val))->GetValue(0));
227  }
228  return std::complex<double>(-999, 0);
