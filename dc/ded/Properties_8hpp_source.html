124 inline const ParticleProperties &FindParticle(std::shared_ptr<PartList> list,
125  pid id) {
126 
127  // position in map
128  int result = -1;
129  for (int i = 0; i < list->size(); ++i) {
130  // if a match is found we skip the rest
131  if (result >= 0)
132  continue;
133 
134  auto it = list->begin();
135  std::advance(it, i);
136  if (it->second.GetId() == id) {
137  result = i;
138  }
139  }
140  if (result < 0) {
141  std::stringstream ss;
142  ss << "FindParticle() | Particle id=" << id << " not found in list!";
143  throw std::runtime_error(ss.str());
144  }
145  auto r = list->begin();
146  std::advance(r, result);
147 
148  return r->second;
149 }

152 inline void ReadParticles(std::shared_ptr<PartList> list,
153  boost::property_tree::ptree pt) {
154 
155  auto particleTree = pt.get_child_optional("ParticleList");
156  if (!particleTree)
157  return;
158 
159  for (auto const &v : particleTree.get()) {
160  auto tmp = ParticleProperties(v.second);
161  auto p = std::make_pair(tmp.GetName(), tmp);
162  auto last = list->insert(p);
163 
164  if (!last.second) {
165  LOG(info) << "ReadParticles() | Particle " << last.first->first
166  << "already exists in list. We overwrite its parameters!";
167  last.first->second = tmp;
168  }
169  tmp = last.first->second;
170 
171  // cparity is optional
172  double cparity = 0.0;
173  try {
174  cparity = tmp.GetQuantumNumber("Cparity");
175  } catch (std::exception &ex) {
176  }
177 
178  LOG(info) << "ReadParticles() | Particle " << tmp.GetName()
179  << " (id=" << tmp.GetId() << ") "
180  << " J(PC)=" << tmp.GetSpinQuantumNumber("Spin") << "("
181  << tmp.GetQuantumNumber("Parity") << cparity << ") "
182  << " mass=" << tmp.GetMass()
183  << " decayType=" << tmp.GetDecayType();
184  }
185 
186  return;
187 }
