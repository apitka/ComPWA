129 inline const ParticleProperties &FindParticle(std::shared_ptr<PartList> list,
130  pid id) {
131  // position in map
132  int result = -1;
133  for (int i = 0; i < list->size(); ++i) {
134  // if a match is found we skip the rest
135  if (result >= 0)
136  continue;
137 
138  auto it = list->begin();
139  std::advance(it, i);
140  if (it->second.GetId() == id) {
141  result = i;
142  }
143  }
144  if (result < 0) {
145  std::stringstream ss;
146  ss << "FindParticle() | Particle id=" << id << " not found in list!";
147  throw std::runtime_error(ss.str());
148  }
149  auto r = list->begin();
150  std::advance(r, result);
151 
152  return r->second;
153 }

156 inline void ReadParticles(std::shared_ptr<PartList> list,
157  const boost::property_tree::ptree &pt) {
158 
159  auto particleTree = pt.get_child_optional("ParticleList");
160  if (!particleTree)
161  return;
162  for (auto const &v : particleTree.get()) {
163  auto tmp = ParticleProperties(v.second);
164  auto p = std::make_pair(tmp.GetName(), tmp);
165  auto last = list->insert(p);
166 
167  if (!last.second) {
168  LOG(info) << "ReadParticles() | Particle " << last.first->first
169  << " already exists in list. We overwrite its parameters!";
170  last.first->second = tmp;
171  }
172  tmp = last.first->second;
173 
174  // cparity is optional
175  double cparity = 0.0;
176  try {
177  cparity = tmp.GetQuantumNumber("Cparity");
178  } catch (std::exception &ex) {
179  }
180 
181  LOG(debug) << "ReadParticles() | Particle " << tmp.GetName()
182  << " (id=" << tmp.GetId() << ") "
183  << " J(PC)=" << tmp.GetSpinQuantumNumber("Spin") << "("
184  << tmp.GetQuantumNumber("Parity") << cparity << ") "
185  << " mass=" << tmp.GetMass()
186  << " decayType=" << tmp.GetDecayType();
187  }
188 
189  return;
190 }

228 inline void UpdateNode(std::shared_ptr<DoubleParameter> p,
229  boost::property_tree::ptree &tr) {
230  for (auto &v : tr.get_child("")) {
231  if (v.first == "Parameter") {
232  std::string nn = v.second.get<std::string>("<xmlattr>.Name");
233  std::string tt = v.second.get<std::string>("<xmlattr>.Type");
234  if (nn == p->GetName()) {
235  // LOG(debug) << "UpdateNode() | Updating node " << v.first
236  // << "." << nn << " to " << p->GetValue();
237  v.second = DoubleParameterSave(*p.get());
238  v.second.put("<xmlattr>.Type", tt);
239  }
240  } else {
241  // Call this function recursively to find multiple matches.
242  UpdateNode(p, v.second);
243  }
244  }
245  return;
246 }

249 inline void UpdateParticleList(std::shared_ptr<PartList> &partL,
250  ParameterList &pars) {
251 
252  boost::property_tree::ptree partTr;
253  partTr.add_child("ParticleList", SaveParticles(partL));
254  // Loop over (double) parameters
255  for (auto i : pars.GetDoubleParameters()) {
256  auto name = i->GetName();
257  // Some default values may not have a name. We skip those.
258  if (name == "")
259  continue;
260  // Search for parameter in tree and update it
261  UpdateNode(i, partTr);
262  }
263 
264  // Create new list from modified tree and assign it to \p partL
265  auto newList = std::make_shared<PartList>();
266  ReadParticles(newList, partTr);
267  partL = newList;
268  return;
269 }
