168  // initialize logging
169  Logging log("DalitzFit-log.txt", boost::log::trivial::debug);
170 
171  // RunManager is supposed to manage all objects of the analysis. It generates
172  // data and starts the fitting procedure.
173  ComPWA::RunManager run;
174 
175  // List with all particle information needed
176  auto partL = std::make_shared<ComPWA::PartList>();
177  ReadParticles(partL, defaultParticleList);
178  ReadParticles(partL, myParticles);
179 
180  //---------------------------------------------------
181  // 1) Create Kinematics object
182  //---------------------------------------------------
183  std::vector<pid> initialState = {443};
184  std::vector<pid> finalState = {22, 111, 111};
185  auto kin =
186  std::make_shared<HelicityKinematics>(partL, initialState, finalState);
187 
188  //---------------------------------------------------
189  // 2) Generate a large phase space sample
190  //---------------------------------------------------
191  auto gen = std::make_shared<ComPWA::Tools::RootGenerator>(partL, kin);
192  run.SetGenerator(gen);
193  std::shared_ptr<Data> phspSample(new Data());
194  run.SetPhspSample(phspSample);
195  run.GeneratePhsp(100000);
196 
197  //---------------------------------------------------
198  // 3) Create intensity from pre-defined model
199  //---------------------------------------------------
200  // Read in model property_tree
201  std::stringstream modelStream;
202  modelStream << amplitudeModel;
203  boost::property_tree::ptree modelTree;
204  boost::property_tree::xml_parser::read_xml(modelStream, modelTree);
205 
206  // Construct intensity class from model string
207  auto intens = IncoherentIntensity::Factory(
208  partL, kin, modelTree.get_child("IncoherentIntensity"));
209 
210  // Pass phsp sample to intensity for normalization.
211  // Convert to dataPoints first.
212  auto phspPoints =
213  std::make_shared<std::vector<dataPoint>>(phspSample->GetDataPoints(kin));
214  intens->SetPhspSample(phspPoints, phspPoints);
215  run.SetAmplitude(intens);
216 
217  //---------------------------------------------------
218  // 4) Generate a data sample given intensity and kinematics
219  //---------------------------------------------------
220  std::shared_ptr<Data> sample(new Data());
221  run.SetData(sample);
222  run.Generate(kin, 1000);
223 
224  //---------------------------------------------------
225  // 5) Fit the model to the data and print the result
226  //---------------------------------------------------
227  ParameterList fitPar;
228  intens->GetParameters(fitPar);
229  // Set start error of 0.05 for parameters, run Minos?
230  setErrorOnParameterList(fitPar, 0.05, false);
231 
232  auto esti = std::make_shared<Estimator::MinLogLH>(
233  kin, intens, sample, phspSample, phspSample, 0, 0);
234 
235  esti->UseFunctionTree(true);
236  LOG(debug) << esti->GetTree()->head()->to_str(25);
237 
238  auto minuitif = new Optimizer::Minuit2::MinuitIF(esti, fitPar);
239  minuitif->SetHesse(true);
240  run.SetOptimizer(std::shared_ptr<Optimizer::Optimizer>(minuitif));
241 
242  // STARTING MINIMIZATION
243  auto result = std::dynamic_pointer_cast<MinuitResult>(run.Fit(fitPar));
244 
245  // Calculate fit fractions
246  std::vector<std::pair<std::string, std::string>> fitComponents;
247  fitComponents.push_back(
248  std::pair<std::string, std::string>("myAmp", "jpsiGammaPiPi"));
249  fitComponents.push_back(
250  std::pair<std::string, std::string>("f2(1270)", "jpsiGammaPiPi"));
251 
252  ParameterList fitFracs =
253  Tools::CalculateFitFractions(kin, intens, phspPoints, fitComponents);
254  // A proper calculation of the fit fraction uncertainty requires
255  // the uncertainty of the fit parameters propagated. We do this
256  // using a numerical approach. Using the covariance matrix
257  // 100 independend sets of fit parameters are generated and the fit fractions
258  // are recalculated. In the end we take the RMS.
259  Tools::CalcFractionError(fitPar, result->GetCovarianceMatrix(), fitFracs, kin,
260  intens, phspPoints, 100, fitComponents);
261 
262  result->SetFitFractions(fitFracs);
263  result->Print();
264 
265  //---------------------------------------------------
266  // 5.1) Save the fit result
267  //---------------------------------------------------
268  std::ofstream ofs("DalitzFit-fitResult.xml");
269  boost::archive::xml_oarchive oa(ofs);
270  oa << BOOST_SERIALIZATION_NVP(result);
271 
272  UpdateParticleList(partL, fitPar);
273  boost::property_tree::ptree ptout;
274  ptout.add_child("ParticleList", SaveParticles(partL));
275  ptout.add_child("IncoherentIntensity", IncoherentIntensity::Save(intens));
276  boost::property_tree::xml_parser::write_xml("DalitzFit-Model.xml", ptout,
277  std::locale());
278  //---------------------------------------------------
279  // 6) Plot data sample and intensity
280  //---------------------------------------------------
281  DalitzPlot pl(kin, "DalitzFit", 100);
282  pl.SetData(sample);
283  pl.SetPhspData(phspSample);
284  pl.SetFitAmp(intens, "", kBlue - 4);
285  pl.Plot();
286  LOG(info) << "Done";
287 
288  return 0;
289 }
