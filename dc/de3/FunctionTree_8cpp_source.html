95 void FunctionTree::createNode(const std::string &name,
96  std::shared_ptr<Strategy> strat,
97  std::string parent, unsigned int dim,
98  bool useVec) {
99  if (dim == 0)
100  throw std::runtime_error("FunctionTree::createNode() | "
101  "Zero dimension! ");
102  if (parent == "" && head_)
103  throw std::runtime_error("FunctionTree::createNode() | "
104  "Head node already exists!");
105 
106  std::vector<std::shared_ptr<AbsParameter>> inter;
107 
108  switch (strat->OutType()) {
109  case ParType::MCOMPLEX: {
110  inter.push_back(std::shared_ptr<AbsParameter>(new MultiComplex(
111  "par" + name,
112  std::vector<std::complex<double>>(dim, std::complex<double>(0., 0.)))));
113  break;
114  }
115  case ParType::MDOUBLE: {
116  inter.push_back(std::shared_ptr<AbsParameter>(
117  new MultiDouble("par" + name, std::vector<double>(dim, 0.))));
118  break;
119  }
120  case ParType::COMPLEX: {
121  std::complex<double> start(0., 0.);
122  if (useVec) {
123  for (unsigned int i = 0; i < dim; i++)
124  inter.push_back(std::shared_ptr<AbsParameter>(
125  new ComplexParameter("par" + name, start)));
126  } else {
127  inter.push_back(std::shared_ptr<AbsParameter>(
128  new ComplexParameter("par" + name, start)));
129  }
130  break;
131  }
132  case ParType::DOUBLE: {
133  double start(0.);
134  if (useVec) {
135  for (unsigned int i = 0; i < dim; i++)
136  inter.push_back(std::shared_ptr<AbsParameter>(
137  new DoubleParameter("par" + name, start)));
138  } else {
139  inter.push_back(std::shared_ptr<AbsParameter>(
140  new DoubleParameter("par" + name, start)));
141  }
142  break;
143  }
144  case ParType::INTEGER: {
145  int start(0);
146  if (useVec) {
147  for (unsigned int i = 0; i < dim; i++)
148  inter.push_back(std::shared_ptr<AbsParameter>(
149  new IntegerParameter("par" + name, start)));
150  } else {
151  inter.push_back(std::shared_ptr<AbsParameter>(
152  new IntegerParameter("par" + name, start)));
153  }
154  break;
155  }
156  case ParType::BOOL: {
157  bool start = false;
158  if (useVec) {
159  for (unsigned int i = 0; i < dim; i++)
160  inter.push_back(std::shared_ptr<AbsParameter>(
161  new BoolParameter("par" + name, start)));
162  } else {
163  inter.push_back(std::shared_ptr<AbsParameter>(
164  new BoolParameter("par" + name, start)));
165  }
166  break;
167  }
168  default: {
169  throw BadParameter("FunctionTree::createNode() | Bad parameter type " +
170  std::to_string(strat->OutType()));
171  }
172  } // end switch
173 
174  std::shared_ptr<TreeNode> newNode, parentNode;
175  if (parent == "") // is this a head node?
176  parentNode = std::shared_ptr<TreeNode>();
177  else
178  parentNode = nodes_.at(parent);
179 
180  if (dim == 1 && !useVec)
181  newNode = std::shared_ptr<TreeNode>(
182  new TreeNode(name, inter[0], strat, parentNode));
183  else
184  newNode =
185  std::shared_ptr<TreeNode>(new TreeNode(name, inter, strat, parentNode));
186 
187  nodes_.insert(
188  std::pair<std::string, std::shared_ptr<TreeNode>>(name, newNode));
189  newNode->linkParents();
190  if (parent == "")
191  head_ = newNode; // if we created a head redirect pointer
192 }

343 bool FunctionTree::sanityCheck() {
344  bool isSane = true;
345  if (!head_)
346  throw std::runtime_error("FunctionTree::sanityCheck() | "
347  "This tree has no head!");
348 
349  // collect all children and parent names
350  std::vector<std::string> childNames, parentNames;
351  getNamesDownward(head_, childNames, parentNames);
352 
353  // check if matches with available nodeNames
354  std::vector<std::string> missedChild, missedParent;
355  for (unsigned int i = 0; i < childNames.size(); i++) {
356  try {
357  nodes_.at(childNames[i]);
358  } catch (std::out_of_range &ex) {
359  missedChild.push_back(childNames[i]);
360  }
361  }
362  for (unsigned int i = 0; i < parentNames.size(); i++) {
363  try {
364  nodes_.at(parentNames[i]);
365  } catch (std::out_of_range &ex) {
366  missedParent.push_back(parentNames[i]);
367  }
368  }
369  if (missedChild.size()) {
370  for (unsigned int i = 0; i < missedChild.size(); i++)
371  LOG(debug) << "This tree misses a child: " << missedChild[i];
372  return false;
373  }
374  if (missedParent.size()) {
375  for (unsigned int i = 0; i < missedParent.size(); i++)
376  LOG(debug) << "This tree misses a parent: " << missedParent[i];
377  return false;
378  }
379 
380  return isSane;
381 }
