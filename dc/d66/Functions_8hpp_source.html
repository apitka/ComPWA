207  if (checkType != out->type())

209 
210  unsigned int nMC = paras.GetNMultiComplex();
211  unsigned int nC = paras.GetNComplex();
212 
213  if (nMC + nC == 0) {
214  // TODO: exception no input
215  return false;
216  }
217 
218  switch (checkType) {
219  case ParType::MCOMPLEX: {
220  // output complex: input must be one multicomplex
221  if (!(nMC == 1)) {
222  // TODO: exception wrong input
223  return false;
224  }
225 
226  for (auto const &multi_complex : paras.GetMultiComplexs()) {
227  unsigned int nElements = multi_complex->GetNValues();
228  // fill MultiDouble parameter
229  std::vector<std::complex<double>> results;
230  results.reserve(nElements);
231  for (unsigned int ele = 0; ele < nElements; ele++) {
232  results.push_back(std::conj(multi_complex->GetValue(ele)));
233  }
234  out = std::shared_ptr<AbsParameter>(
235  new MultiComplex(out->GetName(), results));
236  }
237  break;
238  } // end multi complex
239 
240  case ParType::COMPLEX: {
241  // output complex: input must be a complex
242  if (!(nC == 1)) {
243  // TODO: exception wrong input
244  return false;
245  }
246  out = std::shared_ptr<AbsParameter>(new ComplexParameter(
247  out->GetName(), std::conj(paras.GetComplexParameter(0)->GetValue())));
248  break;
249  } // end double

251  default: {
252  // TODO: exception output partype wrong
