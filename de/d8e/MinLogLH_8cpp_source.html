67 double MinLogLH::controlParameter(ParameterList &minPar) {
68  double lh = 0;
69  if (!_tree) {
70  // Calculate \Sum_{ev} log()
71  double sumLog = 0;
72  // loop over data sample
73  for (unsigned int evt = _firstEvent; evt < _nEvents + _firstEvent; evt++) {
74  dataPoint point;
75  _kin->EventToDataPoint(_dataSample->GetEvent(evt), point);
76  double val = _intens->Intensity(point);
77  sumLog += std::log(val) * point.GetWeight();
78  }
79  lh = (-1) * ((double)_nEvents) / _sumOfWeights * sumLog;
80  } else {
81  _tree->recalculate();
82  std::shared_ptr<DoubleParameter> logLH =
83  std::dynamic_pointer_cast<DoubleParameter>(_tree->head()->getValue());
84  lh = logLH->GetValue();
85  }
86  _nCalls++;
87  return lh; // return -logLH
88 }

117 void MinLogLH::IniLHtree() {
118  LOG(debug) << "MinLogLH::IniLHtree() | Constructing FunctionTree!";
119 
120  // Ensure that a FunctionTree is provided
121  if (!_intens->HasTree())
122  throw std::runtime_error("MinLogLH::IniLHtree() | AmpIntensity does not "
123  "provide a FunctionTree!");
124 
125  _tree = std::shared_ptr<FunctionTree>(new FunctionTree());
126  int sampleSize = _dataSampleList.GetMultiDouble(0)->GetNValues();
127 
128  //-log L = (-1)*N/(\sum_{ev} w_{ev}) \sum_{ev} ...
129  _tree->createHead("LH",
130  std::shared_ptr<Strategy>(new MultAll(ParType::DOUBLE)));
131  _tree->createLeaf("minusOne", -1, "LH");
132  _tree->createLeaf("nEvents", sampleSize, "LH");
133  _tree->createNode("invSumWeights",
134  std::shared_ptr<Strategy>(new Inverse(ParType::DOUBLE)),
135  "LH");
136  _tree->createNode("sumEvents",
137  std::shared_ptr<Strategy>(new AddAll(ParType::DOUBLE)),
138  "LH");
139  _tree->createLeaf("SumOfWeights", _sumOfWeights, "invSumWeights");
140  _tree->createNode("weightLog",
141  std::shared_ptr<Strategy>(new MultAll(ParType::MDOUBLE)),
142  "sumEvents", sampleSize,
143  false); // w_{ev} * log( I_{ev} )
144  _tree->createNode("Log",
145  std::shared_ptr<Strategy>(new LogOf(ParType::MDOUBLE)),
146  "weightLog", sampleSize, false);
147  _tree->insertTree(_intens->GetTree(_kin, _dataSampleList, _phspAccSampleList,
148  _phspSampleList, _kin->GetNVars()),
149  "Log");
150 
151  _tree->recalculate();
152  if (!_tree->sanityCheck()) {
153  throw std::runtime_error("MinLogLH::IniLHtree() | Tree has structural "
154  "problems. Sanity check not passed!");
155  }
156  LOG(debug) << "MinLogLH::IniLHtree() | "
157  "Construction of LH tree finished!";
158  return;
159 }
