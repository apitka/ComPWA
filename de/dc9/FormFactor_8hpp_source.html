23 inline double qSqValue(double sqrtS, double ma, double mb) {
24  double mapb = ma + mb;

40 inline std::complex<double> phspFactor(double sqrtS, double ma, double mb) {
41  double s = sqrtS * sqrtS;
42  std::complex<double> i(0, 1);
43 
44  // == Two types of analytic continuation
45  // 1) Complex sqrt
46  // std::complex<double> rhoOld;
47  // rhoOld = sqrt(std::complex<double>(qSqValue(sqrtS,ma,mb))) /
48  //(8*M_PI*sqrtS); //PDG definition
49  // rhoOld = sqrt(std::complex<double>(qSqValue(sqrtS,ma,mb))) /
50  //(0.5*sqrtS); //BaBar definition
51  // return rhoOld; //complex sqrt
52 
53  /* 2) Correct analytic continuation
54  * proper analytic continuation (PDG 2014 - Resonances (47.2.2))
55  * I'm not sure of this is correct for the case of two different masses ma and
56  * mb.
57  * Furthermore we divide by the factor 16*Pi*Sqrt[s]). This is more or less
58  * arbitrary
59  * and not mentioned in the reference, but it leads to a good agreement
60  * between both
61  * approaches.
62  */
63  std::complex<double> rho;
64  double q = std::sqrt(std::fabs(qSqValue(sqrtS, ma, mb) * 4 / s));
65  if (s <= 0) { // below 0
66  rho = -q / M_PI * std::log(std::fabs((1 + q) / (1 - q)));
67  } else if (0 < s && s <= (ma + mb) * (ma + mb)) { // below threshold
68  rho = (-2.0 * q / M_PI * atan(1 / q)) / (i * 16.0 * M_PI * sqrtS);
69  } else if ((ma + mb) * (ma + mb) < s) { // above threshold
70  rho = (-q / M_PI * log(std::fabs((1 + q) / (1 - q))) + i * q) /
71  (i * 16.0 * M_PI * sqrtS);
72  } else
73  throw std::runtime_error("Kinematics::phspFactor| PhspFactor not "
74  "defined at sqrtS = " +
75  std::to_string((long double)sqrtS));
76 
77 #ifndef NDEBUG
78  if (rho.real() != rho.real() || rho.imag() != rho.imag()) {
79  throw std::runtime_error("Kinematics::phspFactor| Result invalid (NaN)!");
80  }
81 #endif
82 
83  return rho; // correct analytical continuation
84 }

106 inline double FormFactor(double sqrtS, double ma, double mb, double spin,
107  double mesonRadius, std::complex<double> qValue,
108  formFactorType type) {
109  if (mesonRadius == 0)
110  return 1.0; // disable form factors
111  if (type == formFactorType::noFormFactor)
112  return 1.0; // disable form factors
113  if (type == formFactorType::BlattWeisskopf && spin == 0) {
114  return 1.0;
115  }
116 
117  // Form factor for a0(980) used by Crystal Barrel (Phys.Rev.D78-074023)
118  if (type == formFactorType::CrystalBarrel) {
119  if (spin == 0) {
120  double qSq = std::norm(qValue);
121  double alpha = mesonRadius * mesonRadius / 6;
122  return std::exp(-alpha * qSq);
123  } else
124  throw std::runtime_error("Kinematics::FormFactor() | "
125  "Form factors of type " +
126  std::string(formFactorTypeString[type]) +
127  " are implemented for spin 0 only!");
128  }
129 
130  // Blatt-Weisskopt form factors with normalization F(x=mR) = 1.
131  // Reference: S.U.Chung Annalen der Physik 4(1995) 404-430
132  // z = q / (interaction range). For the interaction range we assume
133  // 1/mesonRadius
134  if (type == formFactorType::BlattWeisskopf) {
135  if (spin == 0)
136  return 1;
137  double qSq = std::norm(qValue);
138  double z = qSq * mesonRadius * mesonRadius;
139  /* Events below threshold
140  * What should we do if event is below threshold? Shouldn't really influence
141  * the result
142  * because resonances at threshold don't have spin(?) */
143  z = std::fabs(z);
144 
145  if (spin == 1) {
146  return (sqrt(2 * z / (z + 1)));
147  } else if (spin == 2) {
148  return (sqrt(13 * z * z / ((z - 3) * (z - 3) + 9 * z)));
149  } else if (spin == 3) {
150  return (
151  sqrt(277 * z * z * z / (z * (z - 15) * (z - 15) + 9 * (2 * z - 5))));
152  } else if (spin == 4) {
153  return (sqrt(12746 * z * z * z * z /
154  ((z * z - 45 * z + 105) * (z * z - 45 * z + 105) +
155  25 * z * (2 * z - 21) * (2 * z - 21))));
156  } else
157  throw std::runtime_error(
158  "Kinematics::FormFactor() | Form factors of type " +
159  std::string(formFactorTypeString[type]) +
160  " are implemented for spins up to 4!");
161  }
162  throw std::runtime_error("Kinematics::FormFactor() | Form factor type " +
163  std::to_string((long long int)type) +
164  " not specified!");
165 
166  return 0;
167 }
