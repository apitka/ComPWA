61 double IncoherentIntensity::Intensity(const ComPWA::dataPoint &point) const {
62 
63  // We have to get around the constness of the interface definition.
64  std::vector<std::vector<double>> parameters(_parameters);
65 
66  std::vector<double> normValues(_normValues);
67 
68  if (_intens.size() != parameters.size())
69  parameters = std::vector<std::vector<double>>(_intens.size());
70 
71  if (_intens.size() != normValues.size())
72  normValues = std::vector<double>(_intens.size());
73 
74  double result = 0;
75  for (int i = 0; i < _intens.size(); i++) {
76  std::vector<double> params;
77  _intens.at(i)->GetParametersFast(params);
78  if (parameters.at(i) != params) { // recalculate normalization
79  parameters.at(i) = params;
80  normValues.at(i) =
81  1 / (Tools::Integral(_intens.at(i), _phspSample, phspVolume_));
82  normValues.at(i) *= _intens.at(i)->Strength();
83  }
84  result += _intens.at(i)->Intensity(point) * normValues.at(i);
85  }
86 
87  const_cast<std::vector<std::vector<double>> &>(_parameters) = parameters;
88  const_cast<std::vector<double> &>(_normValues) = normValues;
89 
90  assert(!std::isnan(result) && "IncoherentIntensity::Intensity() | Result is NaN!");
91  assert(!std::isinf(result) && "IncoherentIntensity::Intensity() | Result is inf!");
92 
93  return (Strength() * result);
94 }
