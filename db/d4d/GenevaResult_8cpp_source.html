87 void GenevaResult::PrintCorrelationMatrix(TableFormater *tableCorr) {
88  if (!hasValidCov)
89  return;
90  tableCorr->addColumn(" ", 15); // add empty first column
91  tableCorr->addColumn("GlobalCC", 10); // global correlation coefficient
92 
93  // add columns in correlation matrix
94  for (unsigned int o = 0; o < finalParameters.GetNDouble(); o++) {
95  std::shared_ptr<DoubleParameter> ppp =
96  finalParameters.GetDoubleParameter(o);
97  if (ppp->IsFixed())
98  continue;
99  tableCorr->addColumn(ppp->GetName(), 15);
100  }
101 
102  unsigned int n = 0;
103  tableCorr->header();
104  for (unsigned int o = 0; o < finalParameters.GetNDouble(); o++) {
105  std::shared_ptr<DoubleParameter> ppp =
106  finalParameters.GetDoubleParameter(o);
107  if (ppp->IsFixed())
108  continue;
109  *tableCorr << ppp->GetName();
110  *tableCorr << globalCC.at(n);
111  for (unsigned int t = 0; t < corr.size(); t++) {
112  if (n >= corr.at(0).size()) {
113  *tableCorr << " ";
114  continue;
115  }
116  if (t >= n)
117  *tableCorr << corr.at(n).at(t);
118  else
119  *tableCorr << "";
120  }
121  n++;
122  }
123  tableCorr->footer();
124  return;
