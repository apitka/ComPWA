55  GStartIndividual::GStartIndividual(std::shared_ptr<ComPWA::IEstimator> data, ComPWA::ParameterList list)
56  : GParameterSet(), theData(data), parList(list)
57  {
58  for(std::size_t i=0; i<parList.GetNDouble(); i++) {
59  std::shared_ptr<ComPWA::DoubleParameter> p = parList.GetDoubleParameter(i);
60  if(p->IsFixed()) continue;
61  double val = p->GetValue();
62  double min = -1.79768e+307;
63  double max = 1.79768e+307;
64  double err = val;
65  if(p->HasError()) err = p->GetError();
66  if(p->HasBounds()){
67  min=p->GetMinValue();
68  max=p->GetMaxValue();
69  }
70  std::shared_ptr<GConstrainedDoubleObject> gbd_ptr(
71  new GConstrainedDoubleObject(val, min, max) );
72 
73  // Create a suitable adaptor (sigma=0.1, sigma-adaption=0.5, min sigma=0, max sigma=0,5)
74  std::shared_ptr<GDoubleGaussAdaptor> gdga_ptr(new GDoubleGaussAdaptor(
75  err, 0.5, 0., 3*err));
76  gdga_ptr->setAdaptionThreshold(1); // Adaption parameters are modified after each adaption
77  gdga_ptr->setAdaptionProbability(0.05); // The likelihood for a parameter to be adapted
78 
79  // Register the adaptor with GConstrainedDoubleObject objects
80  gbd_ptr->addAdaptor(gdga_ptr);
81 
82  // Add a GConstrainedDoubleObject object to the collection
83  // gbdc_ptr->push_back(gbd_ptr);
84  // gpoc_ptr->push_back(gbd_ptr);
85  this->push_back(gbd_ptr);
86  //parNames.insert( std::pair<boost::shared_ptr<GConstrainedDoubleObject>,std::string>(gbd_ptr,name[i]) );
87  parNames.push_back(p->GetName());
88  }
89  BOOST_LOG_TRIVIAL(info) << "GStartIndividual::GStartIndividual() | "
90  <<parNames.size()<<" Parameters were added for minimization!";
91  }
