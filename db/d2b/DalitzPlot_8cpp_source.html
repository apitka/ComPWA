124 void DalitzPlot::Fill(std::shared_ptr<Kinematics> kin) {
125  // TODO: reset diagrams here
126 
127  //===== Fill data histograms
128  if (s_data) {
129  for (unsigned int i = 0; i < s_data->GetNEvents(); i++) { // loop over data
130  Event event(s_data->GetEvent(i));
131 
132  double eff = 1.0;
133  if (_correctForEfficiency)
134  eff = event.GetEfficiency();
135  if (eff == 0.0) {
136  LOG(error) << "DalitzPlot::Fill() | Loop over "
137  "data sample: An event with zero efficiency was found! "
138  "This should not happen! We skip it!";
139  continue;
140  }
141  double evWeight = event.GetWeight();
142 
143  dataDiagrams.Fill(kin, event, evWeight * 1 / eff);
144  h_weights.Fill(evWeight * 1 / eff);
145  }
146  _globalScale = dataDiagrams.GetIntegral();
147  }
148 
149  //===== Plot amplitude
150  if (s_phsp) {
151  LOG(info)
152  << "PlotData::plot | Plotting phase space sample and intensity...";
153 
154  double weightsSum = 0.0;
155 
156  /* Loop over all events in phase space sample */
157  progressBar bar(s_phsp->GetNEvents());
158  for (unsigned int i = 0; i < s_phsp->GetNEvents();
159  i++) { // loop over phsp MC
160  bar.nextEvent();
161  Event event(s_phsp->GetEvent(i));
162  double evWeight = event.GetWeight();
163  double eff = 1.0;
164  if (_correctForEfficiency)
165  eff = event.GetEfficiency();
166  if (eff == 0.0) {
167  LOG(error) << "DalitzPlot::Fill() | Loop over "
168  "phsp sample: An event with zero efficiency was found! "
169  "This should not happen! We skip it!";
170  continue;
171  }
172 
173  double evBase = evWeight / eff;
174  weightsSum += evBase;
175 
176  /* Construct DataPoint from Event to check if dataPoint is within the
177  * phase space boundaries */
178  dataPoint point;
179  try {
180  kin->EventToDataPoint(event, point);
181  } catch (BeyondPhsp &ex) { // event outside phase, remove
182  continue;
183  }
184 
185  // Fill diagrams with pure phase space events
186  phspDiagrams.Fill(kin, event, evBase); // scale phsp to data size
187 
188  // Loop over all components that we want to plot
189  for (int t = 0; t < _plotHistograms.size(); t++)
190  _plotHistograms.at(t).Fill(
191  kin, event, _plotComponents.at(t)->Intensity(point) * evBase);
192  }
193 
194  // Scale histograms to match data sample
195  phspDiagrams.Scale(_globalScale / phspDiagrams.GetIntegral());
196  double scale = _globalScale / _plotHistograms.at(0).GetIntegral();
197  _plotHistograms.at(0).Scale(scale);
198 
199  for (int t = 1; t < _plotHistograms.size(); t++) {
200  _plotHistograms.at(t).Scale(scale);
201  }
202  }
203 
204  //===== Plot hit&miss data
205  if (s_hitMiss) {
206  for (unsigned int i = 0; i < s_hitMiss->GetNEvents();
207  i++) { // loop over data
208  Event event(s_hitMiss->GetEvent(i));
209  double eff = 1.0;
210  if (_correctForEfficiency)
211  eff = event.GetEfficiency();
212  if (eff == 0.0) {
213  LOG(error) << "DalitzPlot::Fill() | Loop over "
214  "Hit&Miss sample: An event with zero efficiency was "
215  "found! This should not happen! We skip it!";
216  continue;
217  }
218  double evWeight = event.GetWeight();
219 
220  fitHitMissDiagrams.Fill(kin, event, evWeight * 1 / eff);
221  }
222  }
223 
224  _isFilled = 1;
225 }
