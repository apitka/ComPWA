94 void RootGenerator::Generate(Event &evt) {
95  evt.Clear();
96  const double weight = event.Generate();
97 
98  for (unsigned int t = 0; t < nPart; t++) {
99  TLorentzVector *p = event.GetDecay(t);
100  evt.AddParticle(Particle(p->X(), p->Y(), p->Z(), p->E()));
101  }
102  evt.SetWeight(weight);
103 
104 #ifndef _NDEBUG
105  ComPWA::FourMomentum pFour;
106  for (int i = 0; i < evt.GetNParticles(); i++)
107  pFour += evt.GetParticle(i).GetFourMomentum();
108  if (pFour.GetInvMass() != sqrtS)
109  if (!ComPWA::equal(pFour.GetInvMass(), sqrtS, 100)) {
110  LOG(error) << pFour.GetInvMass() << " - " << sqrtS << " = "
111  << pFour.GetInvMass() - sqrtS;
112  throw std::runtime_error(
113  "RootGenerator::Generate() | Invariant mass of "
114  "all generate particles does not sum up to the mass of the decaying "
115  "particle. "
116  "The difference in larger than two times the numerical precision.");
117  }
118 #endif
119 
120  return;
121 }
