713 inline DoubleParameter
714 DoubleParameterFactory(const boost::property_tree::ptree pt) {
715  DoubleParameter obj;
716 
717  // Require that name and value are provided
718  obj.SetName(pt.get<std::string>("<xmlattr>.Name"));
719  obj.SetValue(pt.get<double>("Value"));
720 
721  // Optional settings
722  if (pt.get_optional<double>("Error")) {
723  obj.SetError(pt.get<double>("Error"));
724  }
725  if (pt.get_optional<double>("ErrorLow")) {
726  if (pt.get_optional<double>("ErrorHigh"))
727  obj.SetError(pt.get<double>("ErrorLow"), pt.get<double>("ErrorHigh"));
728  else
729  throw std::runtime_error("DoubleParameterFactory() | Parameter asymmetic "
730  "error not properly set!");
731  } else if (pt.get_optional<double>("ErrorHigh")) {
732  throw std::runtime_error("DoubleParameterFactory() | Parameter asymmetic "
733  "error not properly set!");
734  } else { /* Do not set a asymmetric errors */
735  }
736 
737  if (pt.get_optional<bool>("Fix"))
738  obj.FixParameter(pt.get<bool>("Fix"));
739 
740  if (pt.get_optional<double>("Min")) {
741  if (pt.get_optional<double>("Max")) {
742  obj.SetRange(pt.get<double>("Min"), pt.get<double>("Max"));
743  } else {
744  throw std::runtime_error(
745  "DoubleParameterFactory() | Parameter range not properly set!");
746  }
747  } else if (pt.get_optional<double>("Max")) {
748  throw std::runtime_error(
749  "DoubleParameterFactory() | Parameter range not properly set!");
750  } else { /* Do not set a parameter range */
751  }
752 
753  return obj;
754 }

764 inline boost::property_tree::ptree DoubleParameterSave(DoubleParameter par) {
765  boost::property_tree::ptree pt;
766  DoubleParameter obj;
767 
768  // Require that name and value are provided
769  pt.put("<xmlattr>.Name", par.GetName());
770  pt.put("Value", par.GetValue());

772  if (par.HasBounds()) {

774  pt.put("Max", par.GetMaxValue());

778  pt.put("Error", par.GetError());
