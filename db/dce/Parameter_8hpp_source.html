704 inline DoubleParameter
705 DoubleParameterFactory(const boost::property_tree::ptree pt) {
706  DoubleParameter obj;
707 
708  // Require that name and value are provided
709  obj.SetName(pt.get<std::string>("<xmlattr>.Name"));
710  obj.SetValue(pt.get<double>("Value"));
711 
712  // Optional settings
713  if (pt.get_optional<double>("Error")) {
714  obj.SetError(pt.get<double>("Error"));
715  }
716  if (pt.get_optional<double>("ErrorLow")) {
717  if (pt.get_optional<double>("ErrorHigh"))
718  obj.SetError(pt.get<double>("ErrorLow"), pt.get<double>("ErrorHigh"));
719  else
720  throw std::runtime_error("DoubleParameterFactory() | Parameter asymmetic "
721  "error not properly set!");
722  } else if (pt.get_optional<double>("ErrorHigh")) {
723  throw std::runtime_error("DoubleParameterFactory() | Parameter asymmetic "
724  "error not properly set!");
725  } else { /* Do not set a asymmetric errors */
726  }
727 
728  if (pt.get_optional<bool>("Fix"))
729  obj.FixParameter(pt.get<bool>("Fix"));
730 
731  if (pt.get_optional<double>("Min")) {
732  if (pt.get_optional<double>("Max")) {
733  obj.SetRange(pt.get<double>("Min"), pt.get<double>("Max"));
734  } else {
735  throw std::runtime_error(
736  "DoubleParameterFactory() | Parameter range not properly set!");
737  }
738  } else if (pt.get_optional<double>("Max")) {
739  throw std::runtime_error(
740  "DoubleParameterFactory() | Parameter range not properly set!");
741  } else { /* Do not set a parameter range */
742  }
743 
744  return obj;
745 }

755  boost::property_tree::ptree pt;

759  pt.put("<xmlattr>.Name", par.GetName());
