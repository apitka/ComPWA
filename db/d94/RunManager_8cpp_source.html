91 bool RunManager::gen(int number, std::shared_ptr<Kinematics> kin, std::shared_ptr<Generator> gen,
92  std::shared_ptr<AmpIntensity> amp,
93  std::shared_ptr<DataReader::Data> data,
94  std::shared_ptr<DataReader::Data> phsp,
95  std::shared_ptr<DataReader::Data> phspTrue) {
96 
97  if (number == 0)
98  return 0;
99 
100  // Doing some checks
101  if (number < 0 && !phsp)
102  throw std::runtime_error("RunManager: gen() negative number of events: " +
103  std::to_string((long double)number) +
104  ". And no phsp sample given!");
105  if (!amp)
106  throw std::runtime_error("RunManager::gen() | Amplitude not valid");
107  if (!gen)
108  throw std::runtime_error("RunManager::gen() | Generator not valid");
109  if (!data)
110  throw std::runtime_error("RunManager::gen() | Sample not valid");
111  if (data->GetNEvents() > 0)
112  throw std::runtime_error("RunManager::gen() | Sample not empty!");
113  if (phspTrue && !phsp)
114  throw std::runtime_error("RunManager::gen() | We have a sample of true"
115  " phsp events, but no phsp sample!");
116  if (phspTrue && phspTrue->GetNEvents() != phsp->GetNEvents())
117  throw std::runtime_error(
118  "RunManager::gen() | We have a sample of true "
119  "phsp events, but the sample size doesn't match that one of "
120  "the phsp sample!");
121 
122  double maxSampleWeight = 1.0;
123  if (phsp)
124  maxSampleWeight = phsp->GetMaxWeight();
125  if (phspTrue && phspTrue->GetMaxWeight() > maxSampleWeight)
126  maxSampleWeight = phspTrue->GetMaxWeight();
127 
128  /* Maximum value for random number generation. We introduce an arbitrary
129  * factor of 5 to make sure that the maximum value is never reached.
130  */
131  double generationMaxValue = 5* maxSampleWeight;
132 
133  unsigned int initialSeed = gen->GetSeed();
134  unsigned int totalCalls = 0;
135 
136  unsigned int limit;
137  if (phsp) {
138  limit = phsp->GetNEvents();
139  generationMaxValue *= Tools::Maximum(kin, amp, phsp) ;
140  } else {
141  limit = 100000000; // set large limit, should never be reached
142  }
143 
144  LOG(trace) << "RunMananger::gen() | Using " << generationMaxValue
145  << " as maximum value of the intensity.";
146 
147  Event evt; // event that we fill into generated sample
148  Event evtTrue; // event that is used to evalutate amplitude
149  progressBar bar(number);
150  if (number <= 0)
151  bar = progressBar(limit);
152  for (unsigned int i = 0; i < limit; i++) {
153  if (phsp && phspTrue) { // phsp and true sample is set
154  evtTrue = phspTrue->GetEvent(i);
155  evt = phsp->GetEvent(i);
156  } else if (phsp) { // phsp sample is set
157  evt = phsp->GetEvent(i);
158  evtTrue = evt;
159  } else { // otherwise generate event
160  gen->Generate(evt);
161  evtTrue = evt;
162  }
163  if (number <= 0)
164  bar.nextEvent();
165 
166  // use reconstructed position for weights
167  double weight = evt.GetWeight();
168 
169  // use true position for amplitude value
170  dataPoint point;
171  try {
172  kin->EventToDataPoint(evtTrue, point);
173  } catch (BeyondPhsp &ex) { // event outside phase, remove
174  continue;
175  }
176 
177  totalCalls++;
178  double ampRnd = gen->GetUniform(0, generationMaxValue);
179  double AMPpdf = amp->Intensity(point);
180 
181  // If maximum of intensity is reached we have to restart the procedure
182  if (generationMaxValue < (AMPpdf * weight)) {
183  LOG(trace) << "RunManager::gen() | Error in HitMiss "
184  "procedure: Maximum value of random number generation "
185  "smaller then amplitude maximum! We raise the maximum "
186  "value and restart generation!";
187  i = 0;
188  bar = progressBar(number);
189  gen->SetSeed(initialSeed);
190  generationMaxValue = 2 * (AMPpdf * weight);
191  data->Clear();
192  totalCalls = 0;
193  continue;
194  }
195 
196  if (ampRnd > (weight * AMPpdf))
197  continue;
198 
199  // Fill event to sample
200  /* reset weights: the weights are taken into account by hit and miss. The
201  * resulting
202  * sample is therefore unweighted */
203  evt.SetWeight(1.); // reset weight
204  evt.SetEfficiency(1.); // reset weight
205  data->PushEvent(evt);
206 
207  if (number > 0)
208  bar.nextEvent();
209  // break if we have a sufficienct number of events
210  if (data->GetNEvents() >= number)
211  i = limit;
212  }
213  if (data->GetNEvents() < number)
214  LOG(error) << "RunManager::gen() | Not able to generate "
215  << number << " events. Phsp sample too small. Current size "
216  "of sample is now " << data->GetNEvents();
217 
218  LOG(info) << "Efficiency of toy MC generation: "
219  << (double)data->GetNEvents() / totalCalls;
220 
221  return true;
222 }

224 bool RunManager::GeneratePhsp(int number) {
225  if (number == 0)
226  return 0;
227  if (!samplePhsp_)
228  throw std::runtime_error("RunManager: generatePhsp() | "
229  "No phase-space sample set");
230  if (samplePhsp_->GetNEvents() > 0)
231  throw std::runtime_error("RunManager: generatePhsp() | "
232  "Dataset not empty! abort!");
233 
234  LOG(info) << "Generating phase-space MC: [" << number << " events] ";
235 
236  progressBar bar(number);
237  for (unsigned int i = 0; i < number; i++) {
238  if (i > 0)
239  i--;
240  Event tmp;
241  gen_->Generate(tmp);
242  double ampRnd = gen_->GetUniform(0, 1);
243  if (ampRnd > tmp.GetWeight())
244  continue;
245 
246  /* Reset weights: weights are taken into account by hit&miss. The
247  * resulting sample is therefore unweighted
248  */
249  tmp.SetWeight(1.);
250 
251  tmp.SetEfficiency(1.);
252  i++;
253  samplePhsp_->PushEvent(tmp); // unfortunatly not thread safe
254  bar.nextEvent();
255  }
256  return true;
257 }
