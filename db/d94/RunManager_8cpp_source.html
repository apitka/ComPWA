77 bool RunManager::gen(int number, std::shared_ptr<Kinematics> kin,
78  std::shared_ptr<Generator> gen,
79  std::shared_ptr<AmpIntensity> amp,
80  std::shared_ptr<DataReader::Data> data,
81  std::shared_ptr<DataReader::Data> phsp,
82  std::shared_ptr<DataReader::Data> phspTrue) {
83 
84  if (number == 0)
85  return 0;
86 
87  // Doing some checks
88  if (number < 0 && !phsp)
89  throw std::runtime_error("RunManager: gen() negative number of events: " +
90  std::to_string((long double)number) +
91  ". And no phsp sample given!");
92  if (!amp)
93  throw std::runtime_error("RunManager::gen() | Amplitude not valid");
94  if (!gen)
95  throw std::runtime_error("RunManager::gen() | Generator not valid");
96  if (!data)
97  throw std::runtime_error("RunManager::gen() | Sample not valid");
98  if (data->GetNEvents() > 0)
99  throw std::runtime_error("RunManager::gen() | Sample not empty!");
100  if (phspTrue && !phsp)
101  throw std::runtime_error("RunManager::gen() | We have a sample of true"
102  " phsp events, but no phsp sample!");
103  if (phspTrue && phspTrue->GetNEvents() != phsp->GetNEvents())
104  throw std::runtime_error(
105  "RunManager::gen() | We have a sample of true "
106  "phsp events, but the sample size doesn't match that one of "
107  "the phsp sample!");
108 
109  double maxSampleWeight = 1.0;
110  if (phsp)
111  maxSampleWeight = phsp->GetMaxWeight();
112  if (phspTrue && phspTrue->GetMaxWeight() > maxSampleWeight)
113  maxSampleWeight = phspTrue->GetMaxWeight();
114 
115  // Maximum value for random number generation. We introduce an arbitrary
116  // factor of 5 to make sure that the maximum value is never reached.
117  double generationMaxValue = 5 * maxSampleWeight;
118 
119  unsigned int initialSeed = gen->GetSeed();
120  unsigned int totalCalls = 0;
121 
122  unsigned int limit;
123  if (phsp) {
124  limit = phsp->GetNEvents();
125  generationMaxValue *= Tools::Maximum(kin, amp, phsp);
126  } else {
127  limit = 100000000; // set large limit, should never be reached
128  }
129 
130  LOG(trace) << "RunMananger::gen() | Using " << generationMaxValue
131  << " as maximum value of the intensity.";
132 
133  Event evt; // event that we fill into generated sample
134  Event evtTrue; // event that is used to evalutate amplitude
135  progressBar bar(number);
136  if (number <= 0)
137  bar = progressBar(limit);
138  for (unsigned int i = 0; i < limit; i++) {
139  if (phsp && phspTrue) { // phsp and true sample is set
140  evtTrue = phspTrue->GetEvent(i);
141  evt = phsp->GetEvent(i);
142  } else if (phsp) { // phsp sample is set
143  evt = phsp->GetEvent(i);
144  evtTrue = evt;
145  } else { // otherwise generate event
146  gen->Generate(evt);
147  evtTrue = evt;
148  }
149  if (number <= 0)
150  bar.nextEvent();
151 
152  // use reconstructed position for weights
153  double weight = evt.GetWeight();
154 
155  // use true position for amplitude value
156  dataPoint point;
157  try {
158  kin->EventToDataPoint(evtTrue, point);
159  } catch (BeyondPhsp &ex) { // event outside phase, remove
160  continue;
161  }
162 
163  totalCalls++;
164  double ampRnd = gen->GetUniform(0, generationMaxValue);
165  double AMPpdf = amp->Intensity(point);
166 
167  // If maximum of intensity is reached we have to restart the procedure
168  if (generationMaxValue < (AMPpdf * weight)) {
169  LOG(trace) << "RunManager::gen() | Error in HitMiss "
170  "procedure: Maximum value of random number generation "
171  "smaller then amplitude maximum! We raise the maximum "
172  "value and restart generation!";
173  i = 0;
174  bar = progressBar(number);
175  gen->SetSeed(initialSeed);
176  generationMaxValue = 2 * (AMPpdf * weight);
177  data->Clear();
178  totalCalls = 0;
179  continue;
180  }
181 
182  if (ampRnd > (weight * AMPpdf))
183  continue;
184 
185  // Fill event to sample
186  // reset weights: the weights are taken into account by hit and miss. The
187  // resulting sample is therefore unweighted
188  evt.SetWeight(1.); // reset weight
189  evt.SetEfficiency(1.); // reset weight
190  data->PushEvent(evt);
191 
192  if (number > 0)
193  bar.nextEvent();
194  // break if we have a sufficienct number of events
195  if (data->GetNEvents() >= number)
196  i = limit;
197  }
198  if (data->GetNEvents() < number) {
199  std::cout << std::endl;
200  LOG(error) << "RunManager::gen() | Not able to generate " << number
201  << " events. Phsp sample too small. Current size "
202  "of sample is now "
203  << data->GetNEvents();
204  }
205 
206  LOG(info) << "Efficiency of toy MC generation: "
207  << (double)data->GetNEvents() / totalCalls;
208 
209  return true;
210 }

212 bool RunManager::GeneratePhsp(int number) {
213  if (number == 0)
214  return 0;
215  if (!samplePhsp_)
216  throw std::runtime_error("RunManager: generatePhsp() | "
217  "No phase-space sample set");
218  if (samplePhsp_->GetNEvents() > 0)
219  throw std::runtime_error("RunManager: generatePhsp() | "
220  "Dataset not empty! abort!");
221 
222  LOG(info) << "Generating phase-space MC: [" << number << " events] ";
223 
224  progressBar bar(number);
225  for (unsigned int i = 0; i < number; i++) {
226  if (i > 0)
227  i--;
228  Event tmp;
229  gen_->Generate(tmp);
230  double ampRnd = gen_->GetUniform(0, 1);
231  if (ampRnd > tmp.GetWeight())
232  continue;
233 
234  // Reset weights: weights are taken into account by hit&miss. The
235  // resulting sample is therefore unweighted
236  tmp.SetWeight(1.);
237 
238  tmp.SetEfficiency(1.);
239  i++;
240  samplePhsp_->PushEvent(tmp); // unfortunatly not thread safe
241  bar.nextEvent();
242  }
243  return true;
244 }
