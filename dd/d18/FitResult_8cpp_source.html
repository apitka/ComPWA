28 double FitResult::shiftAngle(double v) {
29  double originalVal = v;
30  double val = originalVal;
31  double pi = PhysConst::Instance()->FindConstant("Pi").GetValue();
32  while (val > pi)
33  val -= 2 * pi;
34  while (val < -pi)
35  val += 2 * pi;
36  if (val != originalVal)
37  LOG(info) << "shiftAngle(): shifting parameter from " << originalVal
38  << " to " << val << "!";
39  return val;
40 }

92  std::shared_ptr<DoubleParameter> iniPar, outPar, truePar;
93  try {
94  outPar = finalParameters.GetDoubleParameter(o);

103  } catch (BadParameter &bad) {
104  iniPar.reset();
105  }
106  }
107  if (printTrue) {
108  try {
109  truePar = trueParameters.GetDoubleParameter(outPar->GetName());
110  } catch (BadParameter &bad) {
111  iniPar.reset();
112  }
113  }
114 
115  ErrorType errorType = outPar->GetErrorType();
116  bool isFixed = outPar->IsFixed();
117 
118  // Is parameter an angle?
119  bool isAngle = 0;
120  if (outPar->GetName().find("phase") != std::string::npos)
121  isAngle = 1;
122  // ... then shift the value to the domain [-pi;pi]
123  if (isAngle && !isFixed) {
124  outPar->SetValue(shiftAngle(outPar->GetValue()));
125  if (printInitial)
126  iniPar->SetValue(shiftAngle(iniPar->GetValue()));
127  if (printTrue)
128  truePar->SetValue(shiftAngle(truePar->GetValue()));
129  }
130

132  bool isMag = 0;
133  if (outPar->GetName().find("mag") != std::string::npos)
134  isMag = 1;
135  // ... then make sure that it is positive
136  if (isMag && !isFixed) {
137  outPar->SetValue(std::fabs(outPar->GetValue()));
138  if (printInitial && iniPar)
139  iniPar->SetValue(std::fabs(iniPar->GetValue()));
140  if (printTrue && truePar)
141  truePar->SetValue(std::fabs(truePar->GetValue()));
142  }
143 
144  // Print parameter name
145  *tableResult << o << outPar->GetName();
146 
147  // Print initial values
148  if (printInitial) {

151  } else {
152  (*tableResult) << " ";
153  }
154  }
155 
156  // Print final value
157  if (!isFixed)
158  *tableResult << *outPar; // final value
159  else
160  *tableResult << " ";
161 
162  // Print true values
163  if (printTrue) {
164  if (truePar) {
165  *tableResult << *truePar;
166  double pi = PhysConst::Instance()->FindConstant("Pi").GetValue();
167  double pull = (truePar->GetValue() - outPar->GetValue());
168  // Shift pull by 2*pi if that reduces the deviation
169  if (isAngle && !isFixed) {
170  while (pull < 0 && pull < -pi)
171  pull += 2 * pi;
172  while (pull > 0 && pull > pi)
173  pull -= 2 * pi;
174  }
175  if (outPar->HasError()) {
176  if (errorType == ErrorType::ASYM && pull < 0)
177  pull /= outPar->GetErrorLow();
178  else if (errorType == ErrorType::ASYM && pull > 0)
179  pull /= outPar->GetErrorHigh();
180  else
181  pull /= outPar->GetError();
182  }
183  if (!std::isnan(pull))
184  (*tableResult) << pull;
185  else
186  (*tableResult) << " ";
187  } else {
188  (*tableResult) << " "
189  << " ";
190  }
191  }
192  }
193  tableResult->footer();
194 
195  return;
