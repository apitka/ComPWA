84 inline SubSystem SubSystemFactory(const boost::property_tree::ptree pt) {
85  // Read subSystem definition

94  std::vector<std::vector<int>> finalStates;

96  std::vector<std::string> finalStatesNames;

99  stringToVectInt(i.second.get<std::string>("<xmlattr>.FinalState")));
100  finalStatesNames.push_back(i.second.get<std::string>("<xmlattr>.Name"));
101  helicities.push_back(i.second.get<int>("<xmlattr>.Helicity"));
102  }
103  SubSystem subSys(recoilState, finalStates);
104  subSys.SetFinalStatesNames(finalStatesNames);
105  subSys.SetHelicities(helicities);
