187  void SetCouplings(std::vector<Coupling> vC) {
188  if (vC.size() != 2 && vC.size() != 3)
189  throw std::runtime_error(
190  "AmpFlatteRes::SetCouplings() | Vector with "
191  "couplings has a wrong size. We expect either 2 or 3 couplings.");
192 
193  _g = vC;
194 
195  if (_g.size() == 2)
196  _g.push_back(Coupling());
197  // Check if one of the coupling match the final state (_daughterMasses)
198  auto mm = GetDecayMasses();
199  if (mm == std::pair<double, double>(-999, -999))
200  LOG(info)
201  << "AmpFlatteRes::SetCouplings() | Masses of decay products not set. "
202  " Can not determine if correct couplings were set.";
203 
204  bool ok = false;
205  for (auto i : _g) {
206  if (i.GetMassA() == mm.first && i.GetMassB() == mm.second)
207  ok = true;
208  if (i.GetMassB() == mm.first && i.GetMassA() == mm.second)
209  ok = true;
210  }
211  if (!ok)
212  throw std::runtime_error("AmpFlatteRes::SetCouplings() | No couplings "
213  "for the current decay particles set!");
214  }
