140  void SetCouplings(std::vector<Coupling> vC) {
141  if (vC.size() != 2 && vC.size() != 3)
142  throw std::runtime_error(
143  "AmpFlatteRes::SetCouplings() | Vector with "
144  "couplings has a wrong size. We expect either 2 or 3 couplings.");
145 
146  _g = vC;
147 
148  if (_g.size() == 2)
149  _g.push_back(Coupling());
150  // Check if one of the coupling match the final state (_daughterMasses)
151  auto mm = GetDecayMasses();
152  if (mm == std::pair<double, double>(-999, -999))
153  LOG(info)
154  << "AmpFlatteRes::SetCouplings() | Masses of decay products not set. "
155  " Can not determine if correct couplings were set.";
156 
157  bool ok = false;
158  for (auto i : _g) {
159  if (i.GetMassA() == mm.first && i.GetMassB() == mm.second)
160  ok = true;
161  if (i.GetMassB() == mm.first && i.GetMassA() == mm.second)
162  ok = true;
163  }
164  if (!ok)
165  throw std::runtime_error("AmpFlatteRes::SetCouplings() | No couplings "
166  "for the current decay particles set!");
167  }
