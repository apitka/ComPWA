101 bool HelicityKinematics::IsWithinPhsp(const dataPoint &point) const {
102  int subSystemID = 0;
103  int pos = 0;
104  while ((pos + 2) < point.Size()) {
105  auto invMassBounds = GetInvMassBounds(GetSubSystem(subSystemID));
106  if (point.GetValue(pos) < invMassBounds.first ||
107  point.GetValue(pos) > invMassBounds.second)
108  return false;
109  if (point.GetValue(pos + 1) < -1 || point.GetValue(pos + 1) > +1)
110  return false;
111  if (point.GetValue(pos + 2) < 0 || point.GetValue(pos + 2) > 2 * M_PI)
112  return false;
113 
114  pos += 3;
115  subSystemID++;
116  }
117 
118  return true;
119 }

132 std::pair<double, double>
133 HelicityKinematics::CalculateInvMassBounds(const SubSystem sys) const {
134 
135  /* We use the formulae from (PDG2016 Kinematics Fig.47.3). I hope the
136  * generalization to n-body decays is correct.
137  */
138  std::pair<double, double> lim(0, _M);
139  // Sum up masses of all final state particles
140  for (auto j : sys.GetFinalStates())
141  for (auto i : j)
142  lim.first +=
143  PhysConst::Instance()->FindParticle(_finalState.at(i)).GetMass();
144  lim.first *= lim.first;
145 
146  for (auto i : sys.GetRecoilState())
147  lim.second -=
148  PhysConst::Instance()->FindParticle(_finalState.at(i)).GetMass();
149  lim.second *= lim.second;
150 
151  return lim;
152 }

170  double mSpec, double invMassSqA,
171  double invMassSqB) const {
172  // Calculate energy and momentum of m1/m2 in the invMassSqA rest frame
173  double eCms = (invMassSqA + m * m - m2 * m2) / (2 * sqrt(invMassSqA));
174  double pCms = eCms * eCms - m * m;
175  // Calculate energy and momentum of mSpec in the invMassSqA rest frame
176  double eSpecCms =
177  (M * M - mSpec * mSpec - invMassSqA) / (2 * sqrt(invMassSqA));
178  double pSpecCms = eSpecCms * eSpecCms - mSpec * mSpec;
179  double cosAngle =
180  -(invMassSqB - m * m - mSpec * mSpec - 2 * eCms * eSpecCms) /
181  (2 * sqrt(pCms * pSpecCms));
