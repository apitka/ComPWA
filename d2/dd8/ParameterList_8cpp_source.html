74 std::shared_ptr<AbsParameter>
75 ParameterList::GetParameter(const unsigned int i) const {
76  if (i >= GetNParameter())
77  throw BadParameter("ParameterList::GetParameter() | Parameter ID=" +
78  std::to_string(i) + " not in list");
79 
80  unsigned int pos = 0;
81  if (i < vBool_.size())
82  return vBool_.at(i - pos);
83  pos += vBool_.size();
84  if (i < (pos + vInt_.size()))
85  return vInt_.at(i - pos);
86  pos += vInt_.size();
87  if (i < (pos + vDouble_.size()))
88  return vDouble_.at(i - pos);
89  pos += vDouble_.size();
90  if (i < (pos + vComplex_.size()))
91  return vComplex_.at(i - pos);
92  pos += vComplex_.size();
93  if (i < (pos + vMultiDouble_.size()))
94  return vMultiDouble_.at(i - pos);
95  pos += vMultiDouble_.size();
96  if (i < (pos + vMultiComplex_.size()))
97  return vMultiComplex_.at(i - pos);
98  pos += vMultiComplex_.size();
99  if (i < (pos + vMultiUnsignedInteger_.size()))
100  return vMultiUnsignedInteger_.at(i - pos);
101 
102  throw BadParameter("ParameterList::GetParameter() | Parameter ID=" +
103  std::to_string(i) + " not in list");
104 }

144 void ParameterList::RemoveDuplicates() {
145  // We loop over the list of parameters and delete some. Therefore the size
146  // changes during to loop. We enclose GetParameter in try...catch
147  for (int i = 0; i < GetNParameter(); i++) {
148  std::string name;
149  try {
150  name = GetParameter(i)->GetName();
151  } catch (...) {
152  continue;
153  }
154  for (int j = i + 1; j < GetNParameter(); j++) {
155  std::string name2;
156  std::shared_ptr<AbsParameter> delPar;
157  try {
158  delPar = GetParameter(j);
159  name2 = GetParameter(j)->GetName();
160  } catch (...) {
161  continue;
162  }
163 
164  if (name == name2) {
165  auto type = delPar->type();
166  switch (type) {
167  case ParType::BOOL:
168  RemoveBool(j);
169  break;
170  case ParType::INTEGER:
171  RemoveInteger(j - vBool_.size());
172  break;
173  case ParType::DOUBLE:
174  RemoveDouble(j - vBool_.size() - vInt_.size());
175  break;
176  case ParType::COMPLEX:
177  RemoveComplex(j - vBool_.size() - vInt_.size() - vComplex_.size());
178  break;
179  case ParType::MDOUBLE:
180  RemoveMultiDouble(j - vBool_.size() - vInt_.size() -
181  vComplex_.size() - vMultiDouble_.size());
182  break;
183  case ParType::MCOMPLEX:
184  RemoveMultiComplex(j - vBool_.size() - vInt_.size() -
185  vComplex_.size() - vMultiDouble_.size() -
186  vMultiComplex_.size());
187  break;
188  case ParType::MUNSIGNEDINTEGER:
189  RemoveMultiUnsignedInteger(j - vBool_.size() - vInt_.size() -
190  vComplex_.size() - vMultiDouble_.size() -
191  vMultiComplex_.size() -
192  vMultiUnsignedInteger_.size());
193  break;
194  case ParType::UNDEFINED:
195  break;
196  }
197  j--; // decrement if a parameter is removed
198  }
199  }
200  }
201 }

254  auto it = vComplex_.begin();

256  oss << (*it)->GetName() << ": " << (*it)->GetValue() << std::endl;
257  }
258  if (vDouble_.size()) {
259  oss << "Double parameters: " << vDouble_.size() << std::endl;
260  auto it = vDouble_.begin();
261  for (; it != vDouble_.end(); ++it)
262  oss << (*it)->GetName() << ": " << (*it)->GetValue()
263  << " fixed=" << (*it)->IsFixed() << " hasError? "
264  << (*it)->HasError() << std::endl;
265  }
266  if (vInt_.size()) {
267  oss << "Integer parameters: " << vInt_.size() << std::endl;
268  auto it = vInt_.begin();
269  for (; it != vInt_.end(); ++it)
270  oss << (*it)->GetName() << ": " << (*it)->GetValue() << std::endl;
271  }
272  if (vBool_.size()) {
273  oss << "Boolean parameters: " << vBool_.size() << std::endl;
274  auto it = vBool_.begin();
275  for (; it != vBool_.end(); ++it)
276  oss << (*it)->GetName() << ": " << (*it)->GetValue() << std::endl;
277  }
278  if (vMultiComplex_.size()) {
279  oss << "Multi complex parameters: " << vMultiComplex_.size() << std::endl;
280  auto it = vMultiComplex_.begin();
281  for (; it != vMultiComplex_.end(); ++it)
282  oss << (*it)->GetName() << " size=" << (*it)->GetNValues() << std::endl;
283  }
284  if (vMultiUnsignedInteger_.size()) {
285  oss << "Multi unsigned int parameters: " << vMultiUnsignedInteger_.size()
286  << std::endl;
287  auto it = vMultiUnsignedInteger_.begin();
288  for (; it != vMultiUnsignedInteger_.end(); ++it)
289  oss << (*it)->GetName() << " size=" << (*it)->GetNValues() << std::endl;
290  }
291  if (vMultiDouble_.size()) {
292  oss << "Multi double parameters: " << vMultiDouble_.size() << std::endl;
293  auto it = vMultiDouble_.begin();
294  for (; it != vMultiDouble_.end(); ++it)
295  oss << (*it)->GetName() << " size=" << (*it)->GetNValues() << std::endl;
296  }

298  oss << "Parameter List empty!";
