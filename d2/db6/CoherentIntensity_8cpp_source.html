22 double CoherentIntensity::Intensity(const dataPoint &point) const {
23  std::complex<double> result(0., 0.);
24  for (auto i : _seqDecays)
25  result += i->Evaluate(point);
26  assert(!std::isnan(result.real()) && !std::isnan(result.imag()));
27  return Strength() * std::norm(result) * _eff->Evaluate(point);
28 };

71 CoherentIntensity::GetComponent(std::string name) {
72

80  bool found = false;

82  std::vector<std::string> splitNames = splitString(name);
83  auto icIn = std::shared_ptr<AmpIntensity>(this->Clone(name));
84  icIn->Reset(); // delete all existing amplitudes
85  for (auto i : splitNames) {
86  for (auto j : _seqDecays) {
87  if (i == j->GetName()) {
88  std::dynamic_pointer_cast<CoherentIntensity>(icIn)->AddAmplitude(j);
89  found = true;
90  }
91  }
92  }

94  // Nothing found
95  if (!found) {
96  throw std::runtime_error(
97  "CoherentIntensity::GetComponent() | Component " + name +
98  " could not be found in CoherentIntensity " + Name() + ".");
99  }
100 
101  return icIn;
102 }

105 std::shared_ptr<ComPWA::FunctionTree>
106 CoherentIntensity::GetTree(std::shared_ptr<Kinematics> kin,
107  const ComPWA::ParameterList &sample,
108  const ComPWA::ParameterList &phspSample,
109  const ComPWA::ParameterList &toySample,
110  unsigned int nEvtVar, std::string suffix) {
111 
112  unsigned int effId = nEvtVar;
113  unsigned int weightId = nEvtVar + 1;
114  int phspSampleSize = phspSample.GetMultiDouble(0)->GetNValues();
115 
116  std::shared_ptr<MultiDouble> weightPhsp = phspSample.GetMultiDouble(weightId);
117  double sumWeights =
118  std::accumulate(weightPhsp->Begin(), weightPhsp->End(), 0.0);
119  std::shared_ptr<MultiDouble> eff = phspSample.GetMultiDouble(effId);
120 
121  std::shared_ptr<FunctionTree> tr(new FunctionTree());
122 
123  tr->createHead("CoherentIntensity(" + Name() + ")" + suffix,
124  std::shared_ptr<Strategy>(new MultAll(ParType::MDOUBLE)));
125  tr->createLeaf("Strength", _strength,
126  "CoherentIntensity(" + Name() + ")" + suffix);
127  tr->createNode("SumSquared",
128  std::shared_ptr<Strategy>(new AbsSquare(ParType::MDOUBLE)),
129  "CoherentIntensity(" + Name() + ")" + suffix);
130  tr->insertTree(setupBasicTree(kin, sample, toySample), "SumSquared");
131 
132  // Normalization
133  // create a new FunctionTree to make sure that nodes with the same name do
134  // not interfere
135  std::shared_ptr<FunctionTree> trNorm(new FunctionTree());
136  trNorm->createHead("Normalization(" + Name() + ")" + suffix,
137  std::shared_ptr<Strategy>(new Inverse(ParType::DOUBLE)));
138  trNorm->createNode("Integral",
139  std::shared_ptr<Strategy>(new MultAll(ParType::DOUBLE)),
140  "Normalization(" + Name() + ")" + suffix);
141  trNorm->createLeaf("PhspVolume", kin->GetPhspVolume(), "Integral");
142  trNorm->createLeaf("InverseSampleWeights", 1 / ((double)sumWeights),
143  "Integral");
144  trNorm->createNode("Sum",
145  std::shared_ptr<Strategy>(new AddAll(ParType::DOUBLE)),
146  "Integral");
147  trNorm->createNode("IntensityWeighted",
148  std::shared_ptr<Strategy>(new MultAll(ParType::MDOUBLE)),
149  "Sum", phspSampleSize, false);
150  trNorm->createLeaf("Efficiency", eff, "IntensityWeighted");
151  trNorm->createLeaf("EventWeight", weightPhsp, "IntensityWeighted");
152  trNorm->createNode("Intensity",
153  std::shared_ptr<Strategy>(new AbsSquare(ParType::MDOUBLE)),
154  "IntensityWeighted", phspSampleSize,
155  false); //|T_{ev}|^2
156  trNorm->insertTree(setupBasicTree(kin, phspSample, toySample, "_norm"),
157  "Intensity");
158 
159  tr->insertTree(trNorm, "CoherentIntensity(" + Name() + ")" + suffix);
160  return tr;
161 }

163 std::shared_ptr<FunctionTree> CoherentIntensity::setupBasicTree(
164  std::shared_ptr<Kinematics> kin, const ParameterList &sample,
165  const ParameterList &phspSample, std::string suffix) const {
166 
167  int sampleSize = sample.GetMultiDouble(0)->GetNValues();
168  int phspSampleSize = phspSample.GetMultiDouble(0)->GetNValues();
169 
170  if (sampleSize == 0) {
171  LOG(error) << "AmpSumIntensity::setupBasicTree() | "
172  "Data sample empty!";
173  return std::shared_ptr<FunctionTree>();
174  }
175  if (phspSampleSize == 0) {
176  LOG(error) << "AmpSumIntensity::setupBasicTree() | "
177  "Phsp sample empty!";
178  return std::shared_ptr<FunctionTree>();
179  }
180 
181  //------------Setup Tree---------------------
182  std::shared_ptr<FunctionTree> newTree(new FunctionTree());
183 
184  //----Strategies needed
185  std::shared_ptr<AddAll> maddStrat(new AddAll(ParType::MCOMPLEX));
186 
187  newTree->createHead("SumOfAmplitudes" + suffix, maddStrat, sampleSize);
188 
189  for (auto i : _seqDecays) {
190  std::shared_ptr<FunctionTree> resTree =
191  i->GetTree(kin, sample, phspSample, "");
192  if (!resTree->sanityCheck())
193  throw std::runtime_error("AmpSumIntensity::setupBasicTree() | "
194  "Resonance tree didn't pass sanity check!");
195  resTree->recalculate();
196  newTree->insertTree(resTree, "SumOfAmplitudes" + suffix);
197  }
198 
199  LOG(debug) << "AmpSumIntensity::setupBasicTree(): tree constructed!!";
200  return newTree;
201 }
