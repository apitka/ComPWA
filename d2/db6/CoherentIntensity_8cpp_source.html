16 double CoherentIntensity::Intensity(const dataPoint &point) const {
17  std::complex<double> result(0., 0.);
18  for (auto i : _seqDecays)
19  result += i->Evaluate(point);
20  assert(!std::isnan(result.real()) && !std::isnan(result.imag()));
21  return Strength() * std::norm(result) * _eff->Evaluate(point);
22 };

79 CoherentIntensity::GetComponent(std::string name) {
80 
81  // The whole object?
82  if (name == Name()) {
83  LOG(error) << "CoherentIntensity::GetComponent() | You're requesting the "
84  "full object! So just copy it!";
85  return std::shared_ptr<AmpIntensity>();
86  }
87 
88  bool found = false;
89  // Do we want to have a combination of coherentintensities?
90  std::vector<std::string> splitNames = splitString(name);
91  auto icIn = std::shared_ptr<AmpIntensity>(this->Clone(name));
92  icIn->Reset(); // delete all existing amplitudes
93  for (auto i : splitNames) {
94  for (auto j : _seqDecays) {
95  if (i == j->GetName()) {
96  std::dynamic_pointer_cast<CoherentIntensity>(icIn)->AddAmplitude(j);
97  found = true;
98  }
99  }
100  }
101 
102  // Nothing found
103  if (!found) {
104  throw std::runtime_error(
105  "CoherentIntensity::GetComponent() | Component " + name +
106  " could not be found in CoherentIntensity " + Name() + ".");
107  }
108 
109  return icIn;
110 }

113 std::shared_ptr<ComPWA::FunctionTree>
114 CoherentIntensity::GetTree(std::shared_ptr<Kinematics> kin,
115  const ComPWA::ParameterList &sample,
116  const ComPWA::ParameterList &phspSample,
117  const ComPWA::ParameterList &toySample,
118  unsigned int nEvtVar, std::string suffix) {
119 
120  unsigned int effId = nEvtVar;
121  unsigned int weightId = nEvtVar + 1;
122  int phspSampleSize = phspSample.GetMultiDouble(0)->GetNValues();
123 
124  std::shared_ptr<MultiDouble> weightPhsp = phspSample.GetMultiDouble(weightId);
125  double sumWeights =
126  std::accumulate(weightPhsp->Begin(), weightPhsp->End(), 0.0);
127  std::shared_ptr<MultiDouble> eff = phspSample.GetMultiDouble(effId);
128 
129  std::shared_ptr<FunctionTree> tr(new FunctionTree());
130 
131  tr->createHead("CoherentIntensity(" + Name() + ")" + suffix,
132  std::shared_ptr<Strategy>(new MultAll(ParType::MDOUBLE)));
133  tr->createLeaf("Strength", _strength,
134  "CoherentIntensity(" + Name() + ")" + suffix);
135  tr->createNode("SumSquared",
136  std::shared_ptr<Strategy>(new AbsSquare(ParType::MDOUBLE)),
137  "CoherentIntensity(" + Name() + ")" + suffix);
138  tr->insertTree(setupBasicTree(kin, sample, toySample), "SumSquared");
139 
140  // Normalization
141  // create a new FunctionTree to make sure that nodes with the same name do
142  // not interfere
143  std::shared_ptr<FunctionTree> trNorm(new FunctionTree());
144  trNorm->createHead("Normalization(" + Name() + ")" + suffix,
145  std::shared_ptr<Strategy>(new Inverse(ParType::DOUBLE)));
146  trNorm->createNode("Integral",
147  std::shared_ptr<Strategy>(new MultAll(ParType::DOUBLE)),
148  "Normalization(" + Name() + ")" + suffix);
149  trNorm->createLeaf("PhspVolume", kin->GetPhspVolume(), "Integral");
150  trNorm->createLeaf("InverseSampleWeights", 1 / ((double)sumWeights),
151  "Integral");
152  trNorm->createNode("Sum",
153  std::shared_ptr<Strategy>(new AddAll(ParType::DOUBLE)),
154  "Integral");
155  trNorm->createNode("IntensityWeighted",
156  std::shared_ptr<Strategy>(new MultAll(ParType::MDOUBLE)),
157  "Sum", phspSampleSize, false);
158  trNorm->createLeaf("Efficiency", eff, "IntensityWeighted");
159  trNorm->createLeaf("EventWeight", weightPhsp, "IntensityWeighted");
160  trNorm->createNode("Intensity",
161  std::shared_ptr<Strategy>(new AbsSquare(ParType::MDOUBLE)),
162  "IntensityWeighted", phspSampleSize,
163  false); //|T_{ev}|^2
164  trNorm->insertTree(setupBasicTree(kin, phspSample, toySample, "_norm"),
165  "Intensity");
166 
167  tr->insertTree(trNorm, "CoherentIntensity(" + Name() + ")" + suffix);
168  return tr;
169 }

171 std::shared_ptr<FunctionTree> CoherentIntensity::setupBasicTree(
172  std::shared_ptr<Kinematics> kin, const ParameterList &sample,
173  const ParameterList &phspSample, std::string suffix) const {
174 
175  int sampleSize = sample.GetMultiDouble(0)->GetNValues();
176  int phspSampleSize = phspSample.GetMultiDouble(0)->GetNValues();
177 
178  if (sampleSize == 0) {
179  LOG(error) << "AmpSumIntensity::setupBasicTree() | "
180  "Data sample empty!";
181  return std::shared_ptr<FunctionTree>();
182  }
183  if (phspSampleSize == 0) {
184  LOG(error) << "AmpSumIntensity::setupBasicTree() | "
185  "Phsp sample empty!";
186  return std::shared_ptr<FunctionTree>();
187  }
188 
189  //------------Setup Tree---------------------
190  std::shared_ptr<FunctionTree> newTree(new FunctionTree());
191 
192  //----Strategies needed
193  std::shared_ptr<AddAll> maddStrat(new AddAll(ParType::MCOMPLEX));
194 
195  newTree->createHead("SumOfAmplitudes" + suffix, maddStrat, sampleSize);
196 
197  for (auto i : _seqDecays) {
198  std::shared_ptr<FunctionTree> resTree =
199  i->GetTree(kin, sample, phspSample, "");
200  if (!resTree->sanityCheck())
201  throw std::runtime_error("AmpSumIntensity::setupBasicTree() | "
202  "Resonance tree didn't pass sanity check!");
203  resTree->recalculate();
204  newTree->insertTree(resTree, "SumOfAmplitudes" + suffix);
205  }
206 
207  LOG(debug) << "AmpSumIntensity::setupBasicTree(): tree constructed!!";
208  return newTree;
209 }
