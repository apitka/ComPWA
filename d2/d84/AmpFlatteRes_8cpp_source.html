34 AmpFlatteRes::Factory(const boost::property_tree::ptree &pt) {
35  auto obj = std::make_shared<AmpFlatteRes>();
36 
37  std::string name = pt.get<std::string>("DecayParticle.<xmlattr>.Name");
38  LOG(trace) << "AmpFlatteRes::Factory() | Construction of " << name << ".";
39  obj->SetName(name);
40 
41  // All further information on the decay is stored in a ParticleProperty list
42  // which is globally provided by the PhysConst service.
43  auto partProp = PhysConst::Instance()->FindParticle(name);
44 
45  obj->SetMassParameter(
46  std::make_shared<DoubleParameter>(partProp.GetMassPar()));
47 
48  auto decayTr = partProp.GetDecayInfo();
49  if (partProp.GetDecayType() != "flatte")
50  throw std::runtime_error(
51  "AmpFlatteRes::Factory() | Decay type does not match! ");
52 
53  auto spin = partProp.GetSpinQuantumNumber("Spin");
54  obj->SetSpin(spin);
55 
56  auto ffType = formFactorType(decayTr.get<int>("FormFactor.<xmlattr>.Type"));
57  obj->SetFormFactorType(ffType);
58 
59  // Get masses of decay products
60  auto decayProducts = pt.get_child("DecayProducts");
61  if (decayProducts.size() != 2)
62  throw boost::property_tree::ptree_error(
63  "AmpWignerD::Factory() | Expect exactly two decay products (" +
64  std::to_string(decayProducts.size()) + " given)!");
65 
66  auto firstItr = decayProducts.begin();
67  auto secondItr = decayProducts.begin();
68  secondItr++;
69 
70  // Mass of daughter particles needs to be set before the coupling constants
71  std::pair<std::string, std::string> daughterNames(
72  firstItr->second.get<std::string>("<xmlattr>.Name"),
73  secondItr->second.get<std::string>("<xmlattr>.Name"));
74  std::pair<double, double> daughterMasses(
75  PhysConst::Instance()->FindParticle(daughterNames.first).GetMass(),
76  PhysConst::Instance()->FindParticle(daughterNames.second).GetMass());
77 
78  obj->SetDecayMasses(daughterMasses);
79  obj->SetDecayNames(daughterNames);
80 
81  // Read parameters
82  std::vector<Coupling> vC;
83  for (const auto &v : decayTr.get_child("")) {
84  if (v.first != "Parameter")
85  continue;
86  std::string type = v.second.get<std::string>("<xmlattr>.Type");
87  if (type == "Coupling") {
88  vC.push_back(Coupling(v.second));
89  } else if (type == "MesonRadius") {
90  auto mesonRadius = ComPWA::DoubleParameterFactory(v.second);
91  obj->SetMesonRadiusParameter(
92  std::make_shared<DoubleParameter>(mesonRadius));
93  } else {
94  throw std::runtime_error("AmpFlatteRes::Factory() | Parameter of type " +
95  type + " is unknown.");
96  }
97  }
98  obj->SetCouplings(vC);
99 
100  LOG(trace)
101  << "RelativisticBreitWigner::Factory() | Construction of the decay "
102  << partProp.GetName() << " -> " << daughterNames.first << " + "
103  << daughterNames.second;
104 
105  return std::static_pointer_cast<AbstractDynamicalFunction>(obj);
106 }

144 std::complex<double> AmpFlatteRes::dynamicalFunction(
145  double mSq, double mR, double gA, std::complex<double> termA,
146  std::complex<double> termB, std::complex<double> termC) {
147  std::complex<double> i(0, 1);
148  double sqrtS = sqrt(mSq);
149 
150  std::complex<double> denom = std::complex<double>(mR * mR - mSq, 0) +
151  (-1.0) * i * sqrtS * (termA + termB + termC);
152 
153  std::complex<double> result = std::complex<double>(gA, 0) / denom;
154 
155 #ifndef NDEBUG
156  if (std::isnan(result.real()) || std::isnan(result.imag())) {
157  std::cout << "AmpFlatteRes::dynamicalFunction() | " << mR << " " << mSq
158  << " " << termA << " " << termB << " " << termC << std::endl;
159  return 0;
160  }
161 #endif
162 
163  return result;
164 }

166 std::complex<double>
167 AmpFlatteRes::dynamicalFunction(double mSq, double mR, double massA1,
168  double massA2, double gA, double massB1,
169  double massB2, double couplingB, double massC1,
170  double massC2, double couplingC, unsigned int J,
171  double mesonRadius, formFactorType ffType) {
172  std::complex<double> i(0, 1);
173  double sqrtS = sqrt(mSq);
174 
175  // channel A - signal channel
176  std::complex<double> gammaA, qTermA, termA;
177  double barrierA;
178  // break-up momentum
179  barrierA = FormFactor(sqrtS, massA1, massA2, J, mesonRadius, ffType) /
180  FormFactor(mR, massA1, massA2, J, mesonRadius, ffType);
181  // convert coupling to partial width of channel A
182  gammaA = couplingToWidth(mSq, mR, gA, massA1, massA2, J, mesonRadius, ffType);
183  // including the factor qTermA, as suggested by PDG, leads to an amplitude
184  // that doesn't converge.
185  // qTermA = Kinematics::qValue(sqrtS,massA1,massA2) /
186  // Kinematics::qValue(mR,massA1,massA2);
187  qTermA = std::complex<double>(1, 0);
188  termA = gammaA * barrierA * barrierA * std::pow(qTermA, (double)2 * J + 1);
189 
190  // channel B - hidden channel
191  std::complex<double> gammaB, qTermB, termB;
192  double barrierB, gB;
193  // break-up momentum
194  barrierB = FormFactor(sqrtS, massB1, massB2, J, mesonRadius, ffType) /
195  FormFactor(mR, massB1, massB2, J, mesonRadius, ffType);
196  gB = couplingB;
197  // convert coupling to partial width of channel B
198  gammaB = couplingToWidth(mSq, mR, gB, massB1, massB2, J, mesonRadius, ffType);
199  // qTermB = Kinematics::qValue(sqrtS,massB1,massB2) /
200  // Kinematics::qValue(mR,massB1,massB2);
201  qTermB = std::complex<double>(1, 0);
202  termB = gammaB * barrierB * barrierB * std::pow(qTermB, (double)2 * J + 1);
203 
204  // channel C - hidden channel
205  std::complex<double> gammaC, qTermC, termC;
206  double barrierC, gC;
207  if (couplingC != 0.0) {
208  // break-up momentum
209  barrierC = FormFactor(sqrtS, massC1, massC2, J, mesonRadius, ffType) /
210  FormFactor(mR, massC1, massC2, J, mesonRadius, ffType);
211  gC = couplingC;
212  // convert coupling to partial width of channel C
213  gammaC =
214  couplingToWidth(mSq, mR, gC, massC1, massC2, J, mesonRadius, ffType);
215  // qTermC = Kinematics::qValue(sqrtS,massC1,massC2) /
216  // Kinematics::qValue(mR,massC1,massC2);
217  qTermC = std::complex<double>(1, 0);
218  termC = gammaC * barrierC * barrierC * std::pow(qTermC, (double)2 * J + 1);
219  }
220 
221  return dynamicalFunction(mSq, mR, gA, termA, termB, termC);
222 }

260 bool FlatteStrategy::execute(ParameterList &paras,
261  std::shared_ptr<AbsParameter> &out) {
262 #ifndef NDEBUG
263  // Check parameter type
264  if (checkType != out->type())
265  throw(WrongParType("FlatteStrategy::execute() | "
266  "Output parameter is of type " +
267  std::string(ParNames[out->type()]) +
268  " and conflicts with expected type " +
269  std::string(ParNames[checkType])));
270 
271  // How many parameters do we expect?
272  int check_nBool = 0;
273  int check_nInt = 0;
274  int check_nComplex = 0;
275  int check_nDouble = 15;
276  int check_nMDouble = 1;
277  int check_nMComplex = 0;
278 
279  // Check size of parameter list
280  if (paras.GetNBool() != check_nBool)
281  throw(BadParameter("FlatteStrategy::execute() | "
282  "Number of BoolParameters does not match: " +
283  std::to_string(paras.GetNBool()) + " given but " +
284  std::to_string(check_nBool) + " expected."));
285  if (paras.GetNInteger() != check_nInt)
286  throw(BadParameter("FlatteStrategy::execute() | "
287  "Number of IntParameters does not match: " +
288  std::to_string(paras.GetNInteger()) + " given but " +
289  std::to_string(check_nInt) + " expected."));
290  if (paras.GetNDouble() != check_nDouble)
291  throw(BadParameter("FlatteStrategy::execute() | "
292  "Number of DoubleParameters does not match: " +
293  std::to_string(paras.GetNDouble()) + " given but " +
294  std::to_string(check_nDouble) + " expected."));
295  if (paras.GetNComplex() != check_nComplex)
296  throw(BadParameter("FlatteStrategy::execute() | "
297  "Number of ComplexParameters does not match: " +
298  std::to_string(paras.GetNComplex()) + " given but " +
299  std::to_string(check_nComplex) + " expected."));
300  if (paras.GetNMultiDouble() != check_nMDouble)
301  throw(BadParameter("FlatteStrategy::execute() | "
302  "Number of MultiDoubles does not match: " +
303  std::to_string(paras.GetNMultiDouble()) + " given but " +
304  std::to_string(check_nMDouble) + " expected."));
305  if (paras.GetNMultiComplex() != check_nMComplex)
306  throw(BadParameter("FlatteStrategy::execute() | "
307  "Number of MultiComplexes does not match: " +
308  std::to_string(paras.GetNMultiComplex()) +
309  " given but " + std::to_string(check_nMComplex) +
310  " expected."));
311 #endif
312 
313  // invariant masses
314  std::vector<double> mSq = paras.GetMultiDouble(0)->GetValues();
315 
316  std::vector<std::complex<double>> results(mSq.size(),
317  std::complex<double>(0., 0.));
318  // TODO make sure all vectors have to same size
319 
320  // calc function for each point
321  for (unsigned int ele = 0; ele < mSq.size(); ele++) {
322  try {
323  /* Generally we need to add a factor q^{2J+1} to each channel term.
324  * But since Flatte resonances are usually J=0 we neglect it here.*/
325  results.at(ele) = AmpFlatteRes::dynamicalFunction(
326  mSq.at(ele),
327  paras.GetDoubleParameter(0)->GetValue(), // mass
328  paras.GetDoubleParameter(1)->GetValue(), // g1_massA
329  paras.GetDoubleParameter(2)->GetValue(), // g1_massB
330  paras.GetDoubleParameter(3)->GetValue(), // g1
331  paras.GetDoubleParameter(4)->GetValue(), // g2_massA
332  paras.GetDoubleParameter(5)->GetValue(), // g2_massB
333  paras.GetDoubleParameter(6)->GetValue(), // g2
334  paras.GetDoubleParameter(7)->GetValue(), // g3_massA
335  paras.GetDoubleParameter(8)->GetValue(), // g3_massB
336  paras.GetDoubleParameter(9)->GetValue(), // g3
337  paras.GetDoubleParameter(10)->GetValue(), // Spin
338  paras.GetDoubleParameter(11)->GetValue(), // mesonRadius
339  formFactorType(paras.GetDoubleParameter(12)->GetValue()) // ffType
340  );
341  } catch (std::exception &ex) {
342  LOG(error) << "FlatteStrategy::execute() | " << ex.what();
343  throw(std::runtime_error("FlatteStrategy::execute() | "
344  "Evaluation of dynamic function failed!"));
345  }
346  }
347  out =
348  std::shared_ptr<AbsParameter>(new MultiComplex(out->GetName(), results));
349  return true;
350 }

352 void AmpFlatteRes::GetParameters(ParameterList &list) {
353  AbstractDynamicalFunction::GetParameters(list);
354  /* We check of for each parameter if a parameter of the same name exists in
355  * list. If so we check if both are equal and set the local parameter to the
356  * parameter from the list. In this way we connect parameters that occur on
357  * different positions in the amplitude.
358  */
359  std::shared_ptr<DoubleParameter> tmp;
360  for (auto i : _g) {
361  if (i.GetValue() == 0.0)
362  continue;
363  try { // catch BadParameter
364  tmp = list.GetDoubleParameter(i.GetValueParameter()->GetName());
365  try { // catch and throw std::runtime_error due to failed parameter
366  // comparisson
367  if (*tmp == *i.GetValueParameter())
368  i.GetValueParameter() = tmp;
369  } catch (std::exception &ex) {
370  throw;
371  }
372  } catch (BadParameter &ex) {
373  list.AddParameter(i.GetValueParameter());
374  }
375  }
376 
377  try { // catch BadParameter
378  tmp = list.GetDoubleParameter(GetMesonRadiusParameter()->GetName());
379  try { // catch and throw std::runtime_error due to failed parameter
380  // comparisson
381  if (*tmp == *_mesonRadius)
382  SetMesonRadiusParameter(tmp);
383  } catch (std::exception &ex) {
384  throw;
385  }
386  } catch (BadParameter &ex) {
387  list.AddParameter(GetMesonRadiusParameter());
388  }
389 }
