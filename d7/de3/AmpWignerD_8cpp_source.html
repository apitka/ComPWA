36 double AmpWignerD::dynamicalFunction(ComPWA::Spin J, ComPWA::Spin mu,
37  ComPWA::Spin muPrime, double cosTheta) {
38 
39  if ((double)J == 0)
40  return 1.0;
41 
42  assert(!std::isnan(cosTheta));
43 
44  if (cosTheta > 1 || cosTheta < -1)
45  throw std::runtime_error(
46  "AmpWignerD::dynamicalFunction() | "
47  "scattering angle out of range! Datapoint beyond phsp?");
48 
49  double result = QFT::Wigner_d(J, mu, muPrime, acos(cosTheta));
50  assert(!std::isnan(result));
51 
52  // Not quite sure what the correct prefactor is in this case.
53  // double norm = 1/sqrt(2*J.GetSpin()+1);
54  double norm = (2 * J.GetSpin() + 1);
55 
56  return norm * result;
57 }

59 std::complex<double>
60 AmpWignerD::dynamicalFunction(double cosAlpha, double cosBeta, double cosGamma,
61  ComPWA::Spin J, ComPWA::Spin mu,
62  ComPWA::Spin muPrime) {
63  if ((double)J == 0)
64  return 1.0;
65 
66  assert(!std::isnan(cosAlpha));
67  assert(!std::isnan(cosBeta));
68  assert(!std::isnan(cosGamma));
69 
70  std::complex<double> i(0, 1);
71 
72  double tmp = AmpWignerD::dynamicalFunction(J, mu, muPrime, cosBeta);
73  std::complex<double> result =
74  tmp * std::exp(-i * (mu.GetSpin() * acos(cosAlpha) +
75  muPrime.GetSpin() * acos(cosGamma)));
76 
77  assert(!std::isnan(result.real()));
78  assert(!std::isnan(result.imag()));
79 
80  return result;
81 }

147 bool WignerDStrategy::execute(ParameterList &paras,
148  std::shared_ptr<AbsParameter> &out) {
149 #ifndef NDEBUG
150  if (checkType != out->type()) {
151  throw(WrongParType(std::string("Output Type ") + ParNames[out->type()] +
152  std::string(" conflicts expected type ") +
153  ParNames[checkType] + std::string(" of ") + name +
154  " Wigner strat"));
155  return false;
156  }
157 #endif
158 
159  double _inSpin = paras.GetDoubleParameter(0)->GetValue();
160  double _outSpin1 = paras.GetDoubleParameter(1)->GetValue();
161  double _outSpin2 = paras.GetDoubleParameter(2)->GetValue();
162 
163  std::shared_ptr<MultiDouble> _angle = paras.GetMultiDouble(0);
164 
165  std::vector<double> results(_angle->GetNValues(), 0.);
166  for (unsigned int ele = 0; ele < _angle->GetNValues(); ele++) {
167  try {
168  results.at(ele) = AmpWignerD::dynamicalFunction(
169  _inSpin, _outSpin1, _outSpin2, _angle->GetValue(ele));
170  } catch (std::exception &ex) {
171  LOG(error) << "WignerDStrategy::execute() | " << ex.what();
172  throw std::runtime_error("WignerDStrategy::execute() | "
173  "Evaluation of dynamical function failed!");
174  }
175  } // end element loop
176  out = std::shared_ptr<AbsParameter>(new MultiDouble(out->GetName(), results));
177 
178  return true;
179 }
