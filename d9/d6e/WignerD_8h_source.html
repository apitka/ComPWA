39 inline int factorial(int __i) {
40  int f = 1;
41  if ((__i == 0) || (__i == 1))
42  f = 1;
43  else {
44  while (__i > 0) {
45  f = f * __i;
46  __i--;
47  }
48  }
49  return f;
50 }

64 double Clebsch(const double _j1, const double _m1, const double _j2,
65  const double _m2, const double _J, const double _M) {
66  // convert to pure integers (each 2*spin)
67  int j1 = (int)(2. * _j1);
68  int m1 = (int)(2. * _m1);
69  int j2 = (int)(2. * _j2);
70  int m2 = (int)(2. * _m2);
71  int J = (int)(2. * _J);
72  int M = (int)(2. * _M);
73 
74  if ((m1 + m2) != M)
75  return 0.;
76  if (m1 > j1)
77  return 0;
78  if (m2 > j2)
79  return 0;
80 
81  double n0, n1, n2, n3, n4, n5, d0, d1, d2, d3, d4, A, exp;
82  int nu = 0;
83 
84  double sum = 0;
85  while (((d3 = (j1 - j2 - M) / 2 + nu) < 0) ||
86  ((n2 = (j1 - m1) / 2 + nu) < 0)) {
87  nu++;
88  }
89  while (((d1 = (J - j1 + j2) / 2 - nu) >= 0) &&
90  ((d2 = (J + M) / 2 - nu) >= 0) &&
91  ((n1 = (j2 + J + m1) / 2 - nu) >= 0)) {
92  d3 = ((j1 - j2 - M) / 2 + nu);
93  n2 = ((j1 - m1) / 2 + nu);
94  d0 = dfact((double)nu);
95  exp = nu + (j2 + m2) / 2;
96  n0 = (double)pow(-1., exp);
97  sum += ((n0 * dfact(n1) * dfact(n2)) /
98  (d0 * dfact(d1) * dfact(d2) * dfact(d3)));
99  nu++;
100  }
101 
102  if (sum == 0)
103  return 0;
104 
105  n0 = J + 1;
106  n1 = dfact((double)(J + j1 - j2) / 2);
107  n2 = dfact((double)(J - j1 + j2) / 2);
108  n3 = dfact((double)(j1 + j2 - J) / 2);
109  n4 = dfact((double)(J + M) / 2);
110  n5 = dfact((J - M) / 2);
111 
112  d0 = dfact((double)(j1 + j2 + J) / 2 + 1);
113  d1 = dfact((double)(j1 - m1) / 2);
114  d2 = dfact((double)(j1 + m1) / 2);
115  d3 = dfact((double)(j2 - m2) / 2);
116  d4 = dfact((double)(j2 + m2) / 2);
117 
118  A = ((double)(n0 * n1 * n2 * n3 * n4 * n5)) /
119  ((double)(d0 * d1 * d2 * d3 * d4));
120 
121  return sqrt(A) * sum;
122 }

137 double Wigner_d(const double _J, const double _M, const double _N,
138  double _beta) {
139 
140  int J = 2 * _J;
141  int M = 2 * _M;
142  int N = 2 * _N;
143  double beta = _beta;
144 
145  int temp_M, k, k_low, k_hi;
146  double const_term = 0.0, sum_term = 0.0, d = 1.0;
147  int m_p_n, j_p_m, j_p_n, j_m_m, j_m_n;
148  int kmn1, kmn2, jmnk, jmk, jnk;
149  double kk;
150 
151  if (J < 0 || abs(M) > J || abs(N) > J) {
152  std::cerr << std::endl;
153  std::cerr << "d: you have entered an illegal number for J, M, N." << std::endl;
154  std::cerr << "Must follow these rules: J >= 0, abs(M) <= J, and abs(N) <= J."
155  << std::endl;
156  std::cerr << "J = " << J << " M = " << M << " N = " << N << std::endl;
157  return 0.;
158  }
159 
160  if (beta < 0) {
161  beta = fabs(beta);
162  temp_M = M;
163  M = N;
164  N = temp_M;
165  }
166 
167  m_p_n = (M + N) / 2;
168  j_p_m = (J + M) / 2;
169  j_m_m = (J - M) / 2;
170  j_p_n = (J + N) / 2;
171  j_m_n = (J - N) / 2;
172 
173  kk = (double)factorial(j_p_m) * (double)factorial(j_m_m) *
174  (double)factorial(j_p_n) * (double)factorial(j_m_n);
175  const_term = pow((-1.0), (j_p_m)) * sqrt(kk);
176 
177  k_low = MAX(0, m_p_n);
178  k_hi = MIN(j_p_m, j_p_n);
179 
180  for (k = k_low; k <= k_hi; k++) {
181 
182  kmn1 = 2 * k - (M + N) / 2;
183  jmnk = J + (M + N) / 2 - 2 * k;
184  jmk = (J + M) / 2 - k;
185  jnk = (J + N) / 2 - k;
186  kmn2 = k - (M + N) / 2;
187 
188  sum_term +=
189  pow((-1.0), (k)) *
190  ((pow(cos(beta / 2.0), kmn1)) * (pow(sin(beta / 2.0), jmnk))) /
191  (factorial(k) * factorial(jmk) * factorial(jnk) * factorial(kmn2));
192  }
193 
194  d = const_term * sum_term;
195  return d;
196 }

228 void Wigner_d(const Spin &__jmax, double __beta,
229  std::map<Spin, std::map<Spin, std::map<Spin, double>>> &__d) {
230  __d.clear();
231  Spin jmin, one_half = 1 / 2.;
232  if ((double)__jmax == 0.) {
233  __d[0][0][0] = 1.;
234  return;
235  }
236  double cb = cos(__beta);
237  double sb = sin(__beta);
238  // j=1 d's
239  std::map<Spin, std::map<Spin, double>> d1;
240  d1[1][1] = (1 + cb) / 2.;
241  d1[1][0] = -sb / sqrt(2.);
242  d1[1][-1] = (1 - cb) / 2.;
243  d1[0][1] = sb / sqrt(2.);
244  d1[0][0] = cb;
245  d1[0][-1] = -sb / sqrt(2.);
246  d1[-1][1] = (1 - cb) / 2.;
247  d1[-1][0] = sb / sqrt(2.);
248  d1[-1][-1] = (1 + cb) / 2.;
249 
250  if (__jmax.GetDenominator() == 1) { // integral spins
251  __d[0][0][0] = 1.0;
252  if ((double)__jmax == 0.)
253  return;
254  __d[1][1][1] = d1[1][1];
255  __d[1][1][0] = d1[1][0];
256  __d[1][1][-1] = d1[1][-1];
257  __d[1][0][1] = d1[0][1];
258  __d[1][0][0] = d1[0][0];
259  __d[1][0][-1] = d1[0][-1];
260  __d[1][-1][1] = d1[-1][1];
261  __d[1][-1][0] = d1[-1][0];
262  __d[1][-1][-1] = d1[-1][-1];
263  if ((double)__jmax == 1.)
264  return;
265  jmin = 2.;
266  } else { // half-integral spins
267  __d[one_half][one_half][one_half] = cos(__beta / 2.);
268  __d[one_half][one_half][(-1) * (int)one_half] = -sin(__beta / 2.);
269  __d[one_half][(-1) * (int)one_half][one_half] = sin(__beta / 2.);
270  __d[one_half][(-1) * (int)one_half][(-1) * (int)one_half] =
271  cos(__beta / 2.);
272  if (__jmax == one_half)
273  return;
274  jmin = 3 / 2.;
275  }
276 
277  for (Spin j = jmin; j <= __jmax; j++) {
278  for (Spin m = j * (-1); m <= j; m++) {
279  for (Spin n = j * (-1); n <= j; n++) {
280  double djmn = 0.;
281  for (Spin mm = -1; mm <= Spin(1); mm++) {
282  for (Spin nn = -1; nn <= Spin(1); nn++) {
283  djmn += Clebsch((double)j - 1, (double)(m - mm), 1, (double)mm,
284  (double)j, (double)m) *
285  Clebsch((double)j - 1, (double)(n - nn), 1, (double)nn,
286  (double)j, (double)n) *
287  __d[j - Spin(1)][m - mm][n - nn] * d1[mm][nn];
288  }
289  }
290  __d[j][m][n] = djmn;
291  }
292  }
293  }
294 }
