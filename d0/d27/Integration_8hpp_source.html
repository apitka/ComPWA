35  double Integral(int precision = 100) {
36  while (_depth < precision) {
37  // std::cout<< "Current integral approximation: "<<_integral
38  // <<" depth: "<<_depth<<std::endl;
39  Next();
40  _depth *= 2;
41  }
42  return _integral;
43  }
44 
45 protected:
46  const T &_func;
47  std::pair<double, double> _limits;
48  int _depth;
49  double _integral;
50 
51  double Next() {
52  double range = (_limits.second - _limits.first);
53  if (_depth == 1) {
54  _integral =
55  0.5 * (range) * (_func(_limits.first) + _func(_limits.second));
56  } else {
57  double s = 0;
58  int n = 0;
59  double stepSize = 2 * (_limits.second - _limits.first) / (_depth);
60  double x = _limits.first + 0.5 * stepSize;
61  while (x < _limits.second) {
62  s += _func(x);
63  x += stepSize;
64  n++;
65  }
66  if (n <= 0)
67  throw std::runtime_error(
68  "IntegralByQuadrature::Next() | Dividsion by zero!");
69  _integral = 0.5 * (_integral + range * s / n);
70  }
71  return _integral;
72  }
73 };
