35 double shiftAngle(double v) {
36  double originalVal = v;
37  double val = originalVal;
38  while (val > M_PI)
39  val -= 2 * M_PI;
40  while (val < (-1) * M_PI)
41  val += 2 * M_PI;
42  if (val != originalVal)
43  LOG(info) << "shiftAngle() | Shifting parameter from " << originalVal
44  << " to " << val << "!";
45  return val;
46 }

55 std::shared_ptr<FitResult> MinuitIF::exec(ParameterList &par) {
56  LOG(debug) << "MinuitIF::exec() | Start";
57 
58  // Start timing
59  clock_t begin = clock();
60 
61  par.RemoveDuplicates();
62 
63  LOG(debug) << "MinuitIF::exec() | Begin ParameterList::DeepCopy()";
64 
65  ParameterList initialParList;
66  initialParList.DeepCopy(par);
67 
68  MnUserParameters upar;
69  int freePars = 0;
70  for (unsigned int i = 0; i < par.GetNDouble();
71  ++i) { // only doubles for minuit
72  std::shared_ptr<DoubleParameter> actPat = par.GetDoubleParameter(i);
73  // if no error is set or error set to 0 we use a default error,
74  // otherwise minuit treads this parameter as fixed
75  double error = actPat->GetError();
76  if (error <= 0)
77  error = 0.01;
78  if (!actPat->IsFixed() &&
79  actPat->GetName().find("phase") != actPat->GetName().npos)
80  actPat->SetValue(shiftAngle(actPat->GetValue()));
81 
82  if (actPat->HasBounds()) {
83  upar.Add(actPat->GetName(), actPat->GetValue(), error,
84  actPat->GetMinValue(), actPat->GetMaxValue());
85  } else {
86  upar.Add(actPat->GetName(), actPat->GetValue(), error);
87  }
88 
89  if (!actPat->IsFixed())
90  freePars++;
91  if (actPat->IsFixed())
92  upar.Fix(actPat->GetName());
93  }
94 
95  LOG(info) << "MinuitIF::exec() | Number of parameters (free): "
96  << par.GetNDouble() << " (" << freePars << ")";
97 
98  // use MnStrategy class to set all options for the fit
99  MinuitStrategy strat(1); // using default strategy = 1 (medium)
100 
101  try { // try to read xml file for minuit setting
102  std::ifstream ifs("MinuitStrategy.xml");
103  boost::archive::xml_iarchive ia(ifs, boost::archive::no_header);
104  ia >> BOOST_SERIALIZATION_NVP(strat);
105  strat.init(); // update parameters of MnStrategy mother class (IMPORTANT!)
106  ifs.close();
107  LOG(debug) << "Minuit strategy parameters: from MinuitStrategy.xml";
108  } catch (std::exception &ex) {
109  }
110 
111  LOG(debug) << "Gradient number of steps: " << strat.GradientNCycles();
112  LOG(debug) << "Gradient step tolerance: " << strat.GradientStepTolerance();
113  LOG(debug) << "Gradient tolerance: " << strat.GradientTolerance();
114  LOG(debug) << "Hesse number of steps: " << strat.HessianNCycles();
115  LOG(debug) << "Hesse gradient number of steps: "
116  << strat.HessianGradientNCycles();
117  LOG(debug) << "Hesse step tolerance: " << strat.HessianStepTolerance();
118  LOG(debug) << "Hesse G2 tolerance: " << strat.HessianG2Tolerance();
119 
120  // MIGRAD
121  MnMigrad migrad(_myFcn, upar, strat);
122  double maxfcn = 0.0;
123  double tolerance = 0.1;
124 
125  /* From the MINUIT2 Documentation:
126  * Minimize the function MnMigrad()(maxfcn, tolerance)
127  * @param maxfcn : max number of function calls (if = 0) default is
128  * used which is set to 200 + 100 * npar + 5 * npar**2
129  * @param tolerance : value used for terminating iteration procedure.
130  * For example, MIGRAD will stop iterating when edm (expected
131  * distance from minimum) will be: edm < tolerance * 10**-3
132  * Default value of tolerance used is 0.1
133  */
134  LOG(info) << "MinuitIF::exec() | Starting migrad: "
135  "maxCalls="
136  << maxfcn << " tolerance=" << tolerance;
137 
138  FunctionMinimum minMin = migrad(maxfcn, tolerance); //(maxfcn,tolerance)
139 
140  LOG(info) << "MinuitIF::exec() | Migrad finished! "
141  "Minimum is valid = "
142  << minMin.IsValid();
143 
144  // HESSE
145  MnHesse hesse(strat);
146  if (minMin.IsValid() && enableHesse) {
147  LOG(info) << "MinuitIF::exec() | Starting hesse";
148  hesse(_myFcn, minMin); // function minimum minMin is updated by hesse
149  LOG(info) << "MinuitIF::exec() | Hesse finished";
150  } else
151  LOG(info) << "MinuitIF::exec() | Migrad failed to "
152  "find minimum! Skip hesse and minos!";
153 
154  LOG(info) << "MinuitIF::exec() | Minimization finished! "
155  "LH = "
156  << std::setprecision(10) << minMin.Fval();
157 
158  // MINOS
159  MnMinos minos(_myFcn, minMin, strat);
160 
161  // save minimzed values
162  MnUserParameterState minState = minMin.UserState();
163 
164  // ParameterList can be changed by minos. We have to do a deep copy here
165  // to preserve the original parameters at the minimum.
166  ParameterList finalParList;
167  finalParList.DeepCopy(par);
168 
169  // We directly write the central values of the fit result to check if the
170  // parameters change later on
171  std::stringstream resultsOut;
172  resultsOut << "Central values of floating paramters:" << std::endl;
173  for (unsigned int i = 0; i < finalParList.GetNDouble(); ++i) {
174  auto finalPar = finalParList.GetDoubleParameter(i);
175  if (finalPar->IsFixed())
176  continue;
177 
178  // central value
179  double val = minState.Value(finalPar->GetName());
180 
181  // shift to [-pi;pi] if parameter is a phase
182  if (finalPar->GetName().find("phase") != finalPar->GetName().npos)
183  val = shiftAngle(val);
184  finalPar->SetValue(val);
185 
186  resultsOut << finalPar->GetName() << " " << val << std::endl;
187  if (finalPar->GetErrorType() == ErrorType::ASYM) {
188  // Skip minos and fill symmetic errors
189  if (!minMin.IsValid() || !enableMinos) {
190  LOG(info) << "MinuitIF::exec() | Skip Minos "
191  "for parameter "
192  << i << "...";
193  finalPar->SetError(minState.Error(finalPar->GetName()));
194  continue;
195  }
196  // asymmetric errors -> run minos
197  LOG(info) << "MinuitIF::exec() | Run minos "
198  "for parameter "
199  << i << "...";
200  MinosError err = minos.Minos(i);
201  // lower = pair.first, upper= pair.second
202  std::pair<double, double> assymErrors = err();
203  finalPar->SetError(assymErrors.first, assymErrors.second);
204  } else if (finalPar->GetErrorType() == ErrorType::SYM) {
205  // symmetric errors -> migrad/hesse error
206  finalPar->SetError(minState.Error(finalPar->GetName()));
207  } else {
208  throw std::runtime_error(
209  "MinuitIF::exec() | Unknown error type of parameter: " +
210  std::to_string((long long int)finalPar->GetErrorType()));
211  }
212  }
213 
214  //Update the original parameter list
215  for (unsigned int i = 0; i < finalParList.GetNDouble(); ++i) {
216  auto finalPar = finalParList.GetDoubleParameter(i);
217  if( finalPar->IsFixed() ) continue;
218  par.GetDoubleParameter(i)->UpdateParameter(finalPar);
219  }
220 
221  LOG(debug) << "MinuitIF::exec() | " << resultsOut.str();
222 
223  double elapsed = double(clock() - begin) / CLOCKS_PER_SEC;
224 
225  // Create fit result
226  std::shared_ptr<FitResult> result(new MinuitResult(estimator, minMin));
227  result->SetFinalParameters(finalParList);
228  result->SetInitialParameters(initialParList);
229  result->SetTime(elapsed);
230 
231  // update parameters in amplitude
232  // Amplitude::UpdateAmpParameterList(estimator->getAmplitudes(),
233  // finalParList);
234 
235  return result;
236 }
